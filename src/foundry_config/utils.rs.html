24/// Loads the config for the current project workspace or the provided root path
25pub fn load_config_with_root(root: Option<PathBuf>) -> Config {
26    if let Some(root) = root {
27        Config::load_with_root(root)
28    } else {
29        Config::load_with_root(find_project_root_path(None).unwrap())
30    }
31    .sanitized()
32}

34/// Returns the path of the top-level directory of the working git tree. If there is no working
35/// tree, an error is returned.
36pub fn find_git_root_path(relative_to: impl AsRef<Path>) -> eyre::Result<PathBuf> {
37    let path = relative_to.as_ref();
38    let path = std::process::Command::new("git")
39        .args(["rev-parse", "--show-toplevel"])
40        .current_dir(path)
41        .output()
42        .wrap_err_with(|| {
43            format!("Failed detect git root path in current dir: {}", path.display())
44        })?
45        .stdout;
46    let path = std::str::from_utf8(&path)?.trim_end_matches('\n');
47    Ok(PathBuf::from(path))
48}

50/// Returns the root path to set for the project root
51///
52/// traverse the dir tree up and look for a `foundry.toml` file starting at the given path or cwd,
53/// but only until the root dir of the current repo so that
54///
55/// ```text
56/// -- foundry.toml
57///
58/// -- repo
59///   |__ .git
60///   |__sub
61///      |__ [given_path | cwd]
62/// ```
63/// will still detect `repo` as root
64pub fn find_project_root_path(path: Option<&PathBuf>) -> std::io::Result<PathBuf> {
65    let cwd = &std::env::current_dir()?;
66    let cwd = path.unwrap_or(cwd);
67    let boundary = find_git_root_path(cwd)
68        .ok()
69        .filter(|p| !p.as_os_str().is_empty())
70        .unwrap_or_else(|| cwd.clone());
71    let mut cwd = cwd.as_path();
72    // traverse as long as we're in the current git repo cwd
73    while cwd.starts_with(&boundary) {
74        let file_path = cwd.join(Config::FILE_NAME);
75        if file_path.is_file() {
76            return Ok(cwd.to_path_buf())
77        }
78        if let Some(parent) = cwd.parent() {
79            cwd = parent;
80        } else {
81            break
82        }
83    }
84    // no foundry.toml found
85    Ok(boundary)
86}

88/// Returns all [`Remapping`]s contained in the `remappings` str separated by newlines
89///
90/// # Example
91///
92/// ```
93/// use foundry_config::remappings_from_newline;
94/// let remappings: Result<Vec<_>, _> = remappings_from_newline(
95///     r#"
96///              file-ds-test/=lib/ds-test/
97///              file-other/=lib/other/
98///          "#,
99/// )
100/// .collect();
101/// ```
102pub fn remappings_from_newline(
103    remappings: &str,
104) -> impl Iterator<Item = Result<Remapping, RemappingError>> + '_ {
105    remappings.lines().map(|x| x.trim()).filter(|x| !x.is_empty()).map(Remapping::from_str)
106}

108/// Returns the remappings from the given var
109///
110/// Returns `None` if the env var is not set, otherwise all Remappings, See
111/// `remappings_from_newline`
112pub fn remappings_from_env_var(env_var: &str) -> Option<Result<Vec<Remapping>, RemappingError>> {
113    let val = std::env::var(env_var).ok()?;
114    Some(remappings_from_newline(&val).collect())
115}

117/// Converts the `val` into a `figment::Value::Array`
118///
119/// The values should be separated by commas, surrounding brackets are also supported `[a,b,c]`
120pub fn to_array_value(val: &str) -> Result<Value, figment::Error> {
121    let value: Value = match Value::from(val) {
122        Value::String(_, val) => val
123            .trim_start_matches('[')
124            .trim_end_matches(']')
125            .split(',')
126            .map(|s| s.to_string())
127            .collect::<Vec<_>>()
128            .into(),
129        Value::Empty(_, _) => Vec::<Value>::new().into(),
130        val @ Value::Array(_, _) => val,
131        _ => return Err(format!("Invalid value `{val}`, expected an array").into()),
132    };
133    Ok(value)
134}

136/// Returns a list of _unique_ paths to all folders under `root` that contain a `foundry.toml` file
137///
138/// This will also resolve symlinks
139///
140/// # Example
141///
142/// ```no_run
143/// use foundry_config::utils;
144/// let dirs = utils::foundry_toml_dirs("./lib");
145/// ```
146///
147/// for following layout this will return
148/// `["lib/dep1", "lib/dep2"]`
149///
150/// ```text
151/// lib
152/// └── dep1
153/// │   ├── foundry.toml
154/// └── dep2
155///     ├── foundry.toml
156/// ```
157pub fn foundry_toml_dirs(root: impl AsRef<Path>) -> Vec<PathBuf> {
158    walkdir::WalkDir::new(root)
159        .max_depth(1)
160        .into_iter()
161        .filter_map(Result::ok)
162        .filter(|e| e.file_type().is_dir())
163        .filter_map(|e| foundry_compilers::utils::canonicalize(e.path()).ok())
164        .filter(|p| p.join(Config::FILE_NAME).exists())
165        .collect()
166}

168/// Returns a remapping for the given dir
169pub(crate) fn get_dir_remapping(dir: impl AsRef<Path>) -> Option<Remapping> {
170    let dir = dir.as_ref();
171    if let Some(dir_name) = dir.file_name().and_then(|s| s.to_str()).filter(|s| !s.is_empty()) {
172        let mut r = Remapping {
173            context: None,
174            name: format!("{dir_name}/"),
175            path: format!("{}", dir.display()),
176        };
177        if !r.path.ends_with('/') {
178            r.path.push('/')
179        }
180        Some(r)
181    } else {
182        None
183    }
184}

186/// Returns all available `profile` keys in a given `.toml` file
187///
188/// i.e. The toml below would return would return `["default", "ci", "local"]`
189/// ```toml
190/// [profile.default]
191/// ...
192/// [profile.ci]
193/// ...
194/// [profile.local]
195/// ```
196pub fn get_available_profiles(toml_path: impl AsRef<Path>) -> eyre::Result<Vec<String>> {
197    let mut result = vec![Config::DEFAULT_PROFILE.to_string()];
198
199    if !toml_path.as_ref().exists() {
200        return Ok(result)
201    }
202
203    let doc = read_toml(toml_path)?;
204
205    if let Some(Item::Table(profiles)) = doc.as_table().get(Config::PROFILE_SECTION) {
206        for (profile, _) in profiles {
207            let p = profile.to_string();
208            if !result.contains(&p) {
209                result.push(p);
210            }
211        }
212    }
213
214    Ok(result)
215}

217/// Returns a [`toml_edit::Document`] loaded from the provided `path`.
218/// Can raise an error in case of I/O or parsing errors.
219fn read_toml(path: impl AsRef<Path>) -> eyre::Result<DocumentMut> {
220    let path = path.as_ref().to_owned();
221    let doc: DocumentMut = std::fs::read_to_string(path)?.parse()?;
222    Ok(doc)
223}

225/// Deserialize stringified percent. The value must be between 0 and 100 inclusive.
226pub(crate) fn deserialize_stringified_percent<'de, D>(deserializer: D) -> Result<u32, D::Error>
227where
228    D: Deserializer<'de>,
229{
230    let num: U256 = Numeric::deserialize(deserializer)?.into();
231    let num: u64 = num.try_into().map_err(serde::de::Error::custom)?;
232    if num <= 100 {
233        num.try_into().map_err(serde::de::Error::custom)
234    } else {
235        Err(serde::de::Error::custom("percent must be lte 100"))
236    }
237}

239/// Deserialize a `u64` or "max" for `u64::MAX`.
240pub(crate) fn deserialize_u64_or_max<'de, D>(deserializer: D) -> Result<u64, D::Error>
241where
242    D: Deserializer<'de>,
243{
244    #[derive(Deserialize)]
245    #[serde(untagged)]
246    enum Val {
247        Number(u64),
248        String(String),
249    }
250
251    match Val::deserialize(deserializer)? {
252        Val::Number(num) => Ok(num),
253        Val::String(s) if s.eq_ignore_ascii_case("max") => Ok(u64::MAX),
254        Val::String(s) => s.parse::<u64>().map_err(D::Error::custom),
255    }
256}
257
258/// Deserialize a `usize` or "max" for `usize::MAX`.
259pub(crate) fn deserialize_usize_or_max<'de, D>(deserializer: D) -> Result<usize, D::Error>
260where
261    D: Deserializer<'de>,
262{
263    deserialize_u64_or_max(deserializer)?.try_into().map_err(D::Error::custom)
264}

266/// Helper type to parse both `u64` and `U256`
267#[derive(Clone, Copy, Deserialize)]
268#[serde(untagged)]
269pub enum Numeric {
270    /// A [U256] value.
271    U256(U256),
272    /// A `u64` value.
273    Num(u64),
274}
275
276impl From<Numeric> for U256 {
277    fn from(n: Numeric) -> Self {
278        match n {
279            Numeric::U256(n) => n,
280            Numeric::Num(n) => Self::from(n),
281        }
282    }
283}
284
285impl FromStr for Numeric {
286    type Err = String;
287
288    fn from_str(s: &str) -> Result<Self, Self::Err> {
289        if s.starts_with("0x") {
290            U256::from_str_radix(s, 16).map(Numeric::U256).map_err(|err| err.to_string())
291        } else {
292            U256::from_str(s).map(Numeric::U256).map_err(|err| err.to_string())
293        }
294    }
295}

303        EvmVersion::SpuriousDragon => SpecId::SPURIOUS_DRAGON,
304        EvmVersion::Byzantium => SpecId::BYZANTIUM,
305        EvmVersion::Constantinople => SpecId::CONSTANTINOPLE,
306        EvmVersion::Petersburg => SpecId::PETERSBURG,
307        EvmVersion::Istanbul => SpecId::ISTANBUL,
308        EvmVersion::Berlin => SpecId::BERLIN,
309        EvmVersion::London => SpecId::LONDON,
