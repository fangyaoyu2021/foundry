140    /// Runs the test.
141    pub fn run(&self) {
142        // Skip fork tests if the RPC url is not set.
143        if self.fork_block.is_some() && std::env::var_os("ETH_RPC_URL").is_none() {
144            eprintln!("ETH_RPC_URL is not set; skipping");
145            return;
146        }
147
148        let (prj, mut test_cmd) = setup_forge(self.name, self.style.clone());
149
150        // Wipe the default structure.
151        prj.wipe();
152
153        // Clone the external repository.
154        let repo_url = format!("https://github.com/{}/{}.git", self.org, self.name);
155        let root = prj.root().to_str().unwrap();
156        clone_remote(&repo_url, root);
157
158        // Checkout the revision.
159        if self.rev.is_empty() {
160            let mut git = Command::new("git");
161            git.current_dir(root).args(["log", "-n", "1"]);
162            eprintln!("$ {git:?}");
163            let output = git.output().unwrap();
164            if !output.status.success() {
165                panic!("git log failed: {output:?}");
166            }
167            let stdout = String::from_utf8(output.stdout).unwrap();
168            let commit = stdout.lines().next().unwrap().split_whitespace().nth(1).unwrap();
169            panic!("pin to latest commit: {commit}");
170        } else {
171            let mut git = Command::new("git");
172            git.current_dir(root).args(["checkout", self.rev]);
173            eprintln!("$ {git:?}");
174            let status = git.status().unwrap();
175            if !status.success() {
176                panic!("git checkout failed: {status}");
177            }
178        }
179
180        // Run installation command.
181        for install_command in &self.install_commands {
182            let mut install_cmd = Command::new(&install_command[0]);
183            install_cmd.args(&install_command[1..]).current_dir(root);
184            eprintln!("cd {root}; {install_cmd:?}");
185            match install_cmd.status() {
186                Ok(s) => {
187                    eprintln!("\n\n{install_cmd:?}: {s}");
188                    if s.success() {
189                        break;
190                    }
191                }
192                Err(e) => eprintln!("\n\n{install_cmd:?}: {e}"),
193            }
194        }
195
196        // Run the tests.
197        test_cmd.arg("test");
198        test_cmd.args(&self.args);
199        test_cmd.args(["--fuzz-runs=32", "--ffi", "-vvv"]);
200
201        test_cmd.envs(self.envs.iter().map(|(k, v)| (k, v)));
202        if let Some(fork_block) = self.fork_block {
203            test_cmd.env("FOUNDRY_ETH_RPC_URL", crate::rpc::next_http_archive_rpc_endpoint());
204            test_cmd.env("FOUNDRY_FORK_BLOCK_NUMBER", fork_block.to_string());
205        }
206        test_cmd.env("FOUNDRY_INVARIANT_DEPTH", "15");
207
208        test_cmd.assert_non_empty_stdout();
209    }

212/// Initializes a project with `forge init` at the given path.
213///
214/// This should be called after an empty project is created like in
215/// [some of this crate's macros](crate::forgetest_init).
216///
217/// ## Note
218///
219/// This doesn't always run `forge init`, instead opting to copy an already-initialized template
220/// project from a global template path. This is done to speed up tests.
221///
222/// This used to use a `static` [`Lazy`], but this approach does not with `cargo-nextest` because it
223/// runs each test in a separate process. Instead, we use a global lock file to ensure that only one
224/// test can initialize the template at a time.
225pub fn initialize(target: &Path) {
226    eprintln!("initializing {}", target.display());
227
228    let tpath = TEMPLATE_PATH.as_path();
229    pretty_err(tpath, fs::create_dir_all(tpath));
230
231    // Initialize the global template if necessary.
232    let mut lock = crate::fd_lock::new_lock(TEMPLATE_LOCK.as_path());
233    let mut _read = Some(lock.read().unwrap());
234    if fs::read(&*TEMPLATE_LOCK).unwrap() != b"1" {
235        // We are the first to acquire the lock:
236        // - initialize a new empty temp project;
237        // - run `forge init`;
238        // - run `forge build`;
239        // - copy it over to the global template;
240        // Ideally we would be able to initialize a temp project directly in the global template,
241        // but `TempProject` does not currently allow this: https://github.com/foundry-rs/compilers/issues/22
242
243        // Release the read lock and acquire a write lock, initializing the lock file.
244        _read = None;
245
246        let mut write = lock.write().unwrap();
247
248        let mut data = String::new();
249        write.read_to_string(&mut data).unwrap();
250
251        if data != "1" {
252            // Initialize and build.
253            let (prj, mut cmd) = setup_forge("template", foundry_compilers::PathStyle::Dapptools);
254            eprintln!("- initializing template dir in {}", prj.root().display());
255
256            cmd.args(["init", "--force"]).assert_success();
257            // checkout forge-std
258            assert!(Command::new("git")
259                .current_dir(prj.root().join("lib/forge-std"))
260                .args(["checkout", FORGE_STD_REVISION])
261                .output()
262                .expect("failed to checkout forge-std")
263                .status
264                .success());
265            cmd.forge_fuse().args(["build", "--use", SOLC_VERSION]).assert_success();
266
267            // Remove the existing template, if any.
268            let _ = fs::remove_dir_all(tpath);
269
270            // Copy the template to the global template path.
271            pretty_err(tpath, copy_dir(prj.root(), tpath));
272
273            // Update lockfile to mark that template is initialized.
274            write.set_len(0).unwrap();
275            write.seek(std::io::SeekFrom::Start(0)).unwrap();
276            write.write_all(b"1").unwrap();
277        }
278
279        // Release the write lock and acquire a new read lock.
280        drop(write);
281        _read = Some(lock.read().unwrap());
282    }
283
284    eprintln!("- copying template dir from {}", tpath.display());
285    pretty_err(target, fs::create_dir_all(target));
286    pretty_err(target, copy_dir(tpath, target));
287}

289/// Clones a remote repository into the specified directory. Panics if the command fails.
290pub fn clone_remote(repo_url: &str, target_dir: &str) {
291    let mut cmd = Command::new("git");
292    cmd.args(["clone", "--no-tags", "--recursive", "--shallow-submodules"]);
293    cmd.args([repo_url, target_dir]);
294    eprintln!("{cmd:?}");
295    let status = cmd.status().unwrap();
296    if !status.success() {
297        panic!("git clone failed: {status}");
298    }
299    eprintln!();
300}

361/// Setups a new local forge project by cloning and initializing the `RemoteProject`
362///
363/// This will
364///   1. clone the prj, like "transmissions1/solmate"
365///   2. run `forge build`, if configured
366///   3. run additional commands
367///
368/// # Panics
369///
370/// If anything goes wrong during, checkout, build, or other commands are unsuccessful
371pub fn setup_forge_remote(prj: impl Into<RemoteProject>) -> (TestProject, TestCommand) {
372    try_setup_forge_remote(prj).unwrap()
373}
374
375/// Same as `setup_forge_remote` but not panicking
376pub fn try_setup_forge_remote(
377    config: impl Into<RemoteProject>,
378) -> Result<(TestProject, TestCommand)> {
379    let config = config.into();
380    let mut tmp = TempProject::checkout(&config.id).wrap_err("failed to checkout project")?;
381    tmp.project_mut().paths = config.path_style.paths(tmp.root())?;
382
383    let prj = TestProject::with_project(tmp);
384    if config.run_build {
385        let mut cmd = prj.forge_command();
386        cmd.arg("build");
387        cmd.ensure_execute_success().wrap_err("`forge build` unsuccessful")?;
388    }
389    for addon in config.run_commands {
390        debug_assert!(!addon.is_empty());
391        let mut cmd = Command::new(&addon[0]);
392        if addon.len() > 1 {
393            cmd.args(&addon[1..]);
394        }
395        let status = cmd
396            .current_dir(prj.root())
397            .stdout(Stdio::null())
398            .stderr(Stdio::null())
399            .status()
400            .wrap_err_with(|| format!("Failed to execute {addon:?}"))?;
401        eyre::ensure!(status.success(), "Failed to execute command {:?}", addon);
402    }
403
404    let cmd = prj.forge_command();
405    Ok((prj, cmd))
406}

942    pub fn try_execute(&mut self) -> std::io::Result<Output> {
943        eprintln!("executing {:?}", self.cmd);
944        let mut child =
945            self.cmd.stdout(Stdio::piped()).stderr(Stdio::piped()).stdin(Stdio::piped()).spawn()?;

1071/// Extension trait for [`Output`].
1072///
1073/// These function will read the path's content and assert that the process' output matches the
1074/// fixture. Since `forge` commands may emit colorized output depending on whether the current
1075/// terminal is tty, the path argument can be wrapped in [tty_fixture_path()]
1076pub trait OutputExt {
1077    /// Ensure the command wrote the expected data to `stdout`.
1078    fn stdout_matches_content(&self, expected: &str);
1079
1080    /// Ensure the command wrote the expected data to `stdout`.
1081    fn stdout_matches_path(&self, expected_path: impl AsRef<Path>);
1082
1083    /// Ensure the command wrote the expected data to `stderr`.
1084    fn stderr_matches_path(&self, expected_path: impl AsRef<Path>);
1085
1086    /// Returns the stderr as lossy string
1087    fn stderr_lossy(&self) -> String;
1088
1089    /// Returns the stdout as lossy string
1090    fn stdout_lossy(&self) -> String;
1091}
1092
1093/// Patterns to remove from fixtures before comparing output
1094///
1095/// This should strip everything that can vary from run to run, like elapsed time, file paths
1096static IGNORE_IN_FIXTURES: Lazy<Regex> = Lazy::new(|| {
1097    let re = &[
1098        // solc version
1099        r" ?Solc(?: version)? \d+.\d+.\d+",
1100        r" with(?: Solc)? \d+.\d+.\d+",
1101        // solc runs
1102        r"runs: \d+, μ: \d+, ~: \d+",
1103        // elapsed time
1104        r"(?:finished)? ?in .*?s(?: \(.*?s CPU time\))?",
1105        // file paths
1106        r"-->.*\.sol",
1107        r"Location(.|\n)*\.rs(.|\n)*Backtrace",
1108        // other
1109        r"Transaction hash: 0x[0-9A-Fa-f]{64}",
1110    ];
1111    Regex::new(&format!("({})", re.join("|"))).unwrap()
1112});
1113
1114pub fn normalize_output(s: &str) -> String {
1115    let s = s.replace("\r\n", "\n").replace('\\', "/");
1116    IGNORE_IN_FIXTURES.replace_all(&s, "").into_owned()
1117}
1118
1119impl OutputExt for Output {
1120    #[track_caller]
1121    fn stdout_matches_content(&self, expected: &str) {
1122        let out = lossy_string(&self.stdout);
1123        similar_asserts::assert_eq!(normalize_output(&out), normalize_output(expected));
1124    }
1125
1126    #[track_caller]
1127    fn stdout_matches_path(&self, expected_path: impl AsRef<Path>) {
1128        let expected = fs::read_to_string(expected_path).unwrap();
1129        self.stdout_matches_content(&expected);
1130    }
1131
1132    #[track_caller]
1133    fn stderr_matches_path(&self, expected_path: impl AsRef<Path>) {
1134        let expected = fs::read_to_string(expected_path).unwrap();
1135        let err = lossy_string(&self.stderr);
1136        similar_asserts::assert_eq!(normalize_output(&err), normalize_output(&expected));
1137    }
1138
1139    fn stderr_lossy(&self) -> String {
1140        lossy_string(&self.stderr)
1141    }
1142
1143    fn stdout_lossy(&self) -> String {
1144        lossy_string(&self.stdout)
1145    }
1146}

1148/// Returns the fixture path depending on whether the current terminal is tty
1149///
1150/// This is useful in combination with [OutputExt]
1151pub fn tty_fixture_path(path: impl AsRef<Path>) -> PathBuf {
1152    let path = path.as_ref();
1153    if *IS_TTY {
1154        return if let Some(ext) = path.extension().and_then(|s| s.to_str()) {
1155            path.with_extension(format!("tty.{ext}"))
1156        } else {
1157            path.with_extension("tty")
1158        }
1159    }
1160    path.to_path_buf()
1161}

1163/// Return a recursive listing of all files and directories in the given
1164/// directory. This is useful for debugging transient and odd failures in
1165/// integration tests.
1166pub fn dir_list<P: AsRef<Path>>(dir: P) -> Vec<String> {
1167    walkdir::WalkDir::new(dir)
1168        .follow_links(true)
1169        .into_iter()
1170        .map(|result| result.unwrap().path().to_string_lossy().into_owned())
1171        .collect()
1172}

1174fn lossy_string(bytes: &[u8]) -> String {
1175    String::from_utf8_lossy(bytes).replace("\r\n", "\n")
1176}
