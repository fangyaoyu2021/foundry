24/// The aggregated result of a test run.
25#[derive(Clone, Debug)]
26pub struct TestOutcome {
27    /// The results of all test suites by their identifier (`path:contract_name`).
28    ///
29    /// Essentially `identifier => signature => result`.
30    pub results: BTreeMap<String, SuiteResult>,
31    /// Whether to allow test failures without failing the entire test run.
32    pub allow_failure: bool,
33    /// The decoder used to decode traces and logs.
34    ///
35    /// This is `None` if traces and logs were not decoded.
36    ///
37    /// Note that `Address` fields only contain the last executed test case's data.
38    pub last_run_decoder: Option<CallTraceDecoder>,
39    /// The gas report, if requested.
40    pub gas_report: Option<GasReport>,
41}
42
43impl TestOutcome {
44    /// Creates a new test outcome with the given results.
45    pub fn new(results: BTreeMap<String, SuiteResult>, allow_failure: bool) -> Self {
46        Self { results, allow_failure, last_run_decoder: None, gas_report: None }
47    }
48
49    /// Creates a new empty test outcome.
50    pub fn empty(allow_failure: bool) -> Self {
51        Self::new(BTreeMap::new(), allow_failure)
52    }
53
54    /// Returns an iterator over all individual succeeding tests and their names.
55    pub fn successes(&self) -> impl Iterator<Item = (&String, &TestResult)> {
56        self.tests().filter(|(_, t)| t.status == TestStatus::Success)
57    }
58
59    /// Returns an iterator over all individual skipped tests and their names.
60    pub fn skips(&self) -> impl Iterator<Item = (&String, &TestResult)> {
61        self.tests().filter(|(_, t)| t.status == TestStatus::Skipped)
62    }
63
64    /// Returns an iterator over all individual failing tests and their names.
65    pub fn failures(&self) -> impl Iterator<Item = (&String, &TestResult)> {
66        self.tests().filter(|(_, t)| t.status == TestStatus::Failure)
67    }
68
69    /// Returns an iterator over all individual tests and their names.
70    pub fn tests(&self) -> impl Iterator<Item = (&String, &TestResult)> {
71        self.results.values().flat_map(|suite| suite.tests())
72    }
73
74    /// Flattens the test outcome into a list of individual tests.
75    // TODO: Replace this with `tests` and make it return `TestRef<'_>`
76    pub fn into_tests_cloned(&self) -> impl Iterator<Item = SuiteTestResult> + '_ {
77        self.results
78            .iter()
79            .flat_map(|(file, suite)| {
80                suite
81                    .test_results
82                    .iter()
83                    .map(move |(sig, result)| (file.clone(), sig.clone(), result.clone()))
84            })
85            .map(|(artifact_id, signature, result)| SuiteTestResult {
86                artifact_id,
87                signature,
88                result,
89            })
90    }
91
92    /// Flattens the test outcome into a list of individual tests.
93    pub fn into_tests(self) -> impl Iterator<Item = SuiteTestResult> {
94        self.results
95            .into_iter()
96            .flat_map(|(file, suite)| {
97                suite.test_results.into_iter().map(move |t| (file.clone(), t))
98            })
99            .map(|(artifact_id, (signature, result))| SuiteTestResult {
100                artifact_id,
101                signature,
102                result,
103            })
104    }
105
106    /// Returns the number of tests that passed.
107    pub fn passed(&self) -> usize {
108        self.successes().count()
109    }
110
111    /// Returns the number of tests that were skipped.
112    pub fn skipped(&self) -> usize {
113        self.skips().count()
114    }
115
116    /// Returns the number of tests that failed.
117    pub fn failed(&self) -> usize {
118        self.failures().count()
119    }
120
121    /// Sums up all the durations of all individual test suites.
122    ///
123    /// Note that this is not necessarily the wall clock time of the entire test run.
124    pub fn total_time(&self) -> Duration {
125        self.results.values().map(|suite| suite.duration).sum()
126    }
127
128    /// Formats the aggregated summary of all test suites into a string (for printing).
129    pub fn summary(&self, wall_clock_time: Duration) -> String {
130        let num_test_suites = self.results.len();
131        let suites = if num_test_suites == 1 { "suite" } else { "suites" };
132        let total_passed = self.passed();
133        let total_failed = self.failed();
134        let total_skipped = self.skipped();
135        let total_tests = total_passed + total_failed + total_skipped;
136        format!(
137            "\nRan {} test {} in {:.2?} ({:.2?} CPU time): {} tests passed, {} failed, {} skipped ({} total tests)",
138            num_test_suites,
139            suites,
140            wall_clock_time,
141            self.total_time(),
142            total_passed.green(),
143            total_failed.red(),
144            total_skipped.yellow(),
145            total_tests
146        )
147    }
148
149    /// Checks if there are any failures and failures are disallowed.
150    pub fn ensure_ok(&self) -> eyre::Result<()> {
151        let outcome = self;
152        let failures = outcome.failures().count();
153        if outcome.allow_failure || failures == 0 {
154            return Ok(());
155        }
156
157        if !shell::verbosity().is_normal() {
158            // TODO: Avoid process::exit
159            std::process::exit(1);
160        }
161
162        shell::println("")?;
163        shell::println("Failing tests:")?;
164        for (suite_name, suite) in outcome.results.iter() {
165            let failed = suite.failed();
166            if failed == 0 {
167                continue;
168            }
169
170            let term = if failed > 1 { "tests" } else { "test" };
171            shell::println(format!("Encountered {failed} failing {term} in {suite_name}"))?;
172            for (name, result) in suite.failures() {
173                shell::println(result.short_result(name))?;
174            }
175            shell::println("")?;
176        }
177        let successes = outcome.passed();
178        shell::println(format!(
179            "Encountered a total of {} failing tests, {} tests succeeded",
180            failures.to_string().red(),
181            successes.to_string().green()
182        ))?;
183
184        // TODO: Avoid process::exit
185        std::process::exit(1);
186    }
187}

189/// A set of test results for a single test suite, which is all the tests in a single contract.

236    pub fn failed(&self) -> usize {
237        self.failures().count()
238    }
239
240    /// Iterator over all tests and their names
241    pub fn tests(&self) -> impl Iterator<Item = (&String, &TestResult)> {
242        self.test_results.iter()
243    }
244
245    /// Whether this test suite is empty.
246    pub fn is_empty(&self) -> bool {
247        self.test_results.is_empty()
248    }
249
250    /// The number of tests in this test suite.
251    pub fn len(&self) -> usize {
252        self.test_results.len()
253    }
254
255    /// Sums up all the durations of all individual tests in this suite.
256    ///
257    /// Note that this is not necessarily the wall clock time of the entire test suite.
258    pub fn total_time(&self) -> Duration {

261
262    /// Returns the summary of a single test suite.
263    pub fn summary(&self) -> String {
264        let failed = self.failed();
265        let result = if failed == 0 { "ok".green() } else { "FAILED".red() };
266        format!(
267            "Suite result: {}. {} passed; {} failed; {} skipped; finished in {:.2?} ({:.2?} CPU time)",
268            result,
269            self.passed().green(),
270            failed.red(),
271            self.skipped().yellow(),
272            self.duration,
273            self.total_time(),
274        )
275    }

338/// The result of an executed test.
339#[derive(Clone, Debug, Default, Serialize, Deserialize)]
340pub struct TestResult {
341    /// The test status, indicating whether the test case succeeded, failed, or was marked as
342    /// skipped. This means that the transaction executed properly, the test was marked as
343    /// skipped with vm.skip(), or that there was a revert and that the test was expected to
344    /// fail (prefixed with `testFail`)
345    pub status: TestStatus,
346
347    /// If there was a revert, this field will be populated. Note that the test can
348    /// still be successful (i.e self.success == true) when it's expected to fail.
349    pub reason: Option<String>,
350
351    /// Minimal reproduction test case for failing test
352    pub counterexample: Option<CounterExample>,
353
354    /// Any captured & parsed as strings logs along the test's execution which should
355    /// be printed to the user.
356    pub logs: Vec<Log>,
357
358    /// What kind of test this was
359    pub kind: TestKind,
360
361    /// Traces
362    #[serde(skip)]
363    pub traces: Traces,
364
365    /// Additional traces to use for gas report.
366    #[serde(skip)]
367    pub gas_report_traces: Vec<Vec<CallTraceArena>>,
368
369    /// Raw coverage info
370    #[serde(skip)]
371    pub coverage: Option<HitMaps>,
372
373    /// Labeled addresses
374    pub labeled_addresses: HashMap<Address, String>,
375
376    pub duration: Duration,
377
378    /// pc breakpoint char map
379    pub breakpoints: Breakpoints,
380}
381
382impl fmt::Display for TestResult {
383    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
384        match self.status {
385            TestStatus::Success => "[PASS]".green().fmt(f),
386            TestStatus::Skipped => "[SKIP]".yellow().fmt(f),
387            TestStatus::Failure => {
388                let mut s = String::from("[FAIL. Reason: ");
389
390                let reason = self.reason.as_deref().unwrap_or("assertion failed");
391                s.push_str(reason);
392
393                if let Some(counterexample) = &self.counterexample {
394                    match counterexample {
395                        CounterExample::Single(ex) => {
396                            write!(s, "; counterexample: {ex}]").unwrap();
397                        }
398                        CounterExample::Sequence(sequence) => {
399                            s.push_str("]\n\t[Sequence]\n");
400                            for ex in sequence {
401                                writeln!(s, "\t\t{ex}").unwrap();
402                            }
403                        }
404                    }
405                } else {
406                    s.push(']');
407                }
408
409                s.red().fmt(f)
410            }
411        }
412    }
413}

457
458    /// Returns the result for single test. Merges execution results (logs, labeled addresses,
459    /// traces and coverages) in initial setup results.
460    pub fn single_result(
461        mut self,
462        success: bool,
463        reason: Option<String>,
464        raw_call_result: RawCallResult,
465    ) -> Self {
466        self.kind =
467            TestKind::Unit { gas: raw_call_result.gas_used.wrapping_sub(raw_call_result.stipend) };
468
469        // Record logs, labels, traces and merge coverages.
470        self.logs.extend(raw_call_result.logs);
471        self.labeled_addresses.extend(raw_call_result.labels);
472        self.traces.extend(raw_call_result.traces.map(|traces| (TraceKind::Execution, traces)));
473        self.merge_coverages(raw_call_result.coverage);
474
475        self.status = match success {
476            true => TestStatus::Success,
477            false => TestStatus::Failure,
478        };
479        self.reason = reason;
480        self.breakpoints = raw_call_result.cheatcodes.map(|c| c.breakpoints).unwrap_or_default();
481        self.duration = Duration::default();
482        self.gas_report_traces = Vec::new();
483        self
484    }
485
486    /// Returns the result for a fuzzed test. Merges fuzz execution results (logs, labeled
487    /// addresses, traces and coverages) in initial setup results.
488    pub fn fuzz_result(mut self, result: FuzzTestResult) -> Self {
489        self.kind = TestKind::Fuzz {
490            median_gas: result.median_gas(false),
491            mean_gas: result.mean_gas(false),
492            first_case: result.first_case,
493            runs: result.gas_by_case.len(),
494        };
495
496        // Record logs, labels, traces and merge coverages.
497        self.logs.extend(result.logs);
498        self.labeled_addresses.extend(result.labeled_addresses);
499        self.traces.extend(result.traces.map(|traces| (TraceKind::Execution, traces)));
500        self.merge_coverages(result.coverage);
501
502        self.status = match result.success {
503            true => TestStatus::Success,
504            false => TestStatus::Failure,
505        };
506        self.reason = result.reason;
507        self.counterexample = result.counterexample;
508        self.duration = Duration::default();
509        self.gas_report_traces = result.gas_report_traces.into_iter().map(|t| vec![t]).collect();
510        self.breakpoints = result.breakpoints.unwrap_or_default();
511        self
512    }

514    /// Returns the skipped result for invariant test.

521    /// Returns the fail result for replayed invariant test.
522    pub fn invariant_replay_fail(
523        mut self,
524        replayed_entirely: bool,
525        invariant_name: &String,
526        call_sequence: Vec<BaseCounterExample>,
527    ) -> Self {
528        self.kind = TestKind::Invariant { runs: 1, calls: 1, reverts: 1 };
529        self.status = TestStatus::Failure;
530        self.reason = if replayed_entirely {
531            Some(format!("{invariant_name} replay failure"))
532        } else {
533            Some(format!("{invariant_name} persisted failure revert"))

539    /// Returns the fail result for invariant test setup.
540    pub fn invariant_setup_fail(mut self, e: Report) -> Self {
541        self.kind = TestKind::Invariant { runs: 0, calls: 0, reverts: 0 };
542        self.status = TestStatus::Failure;
543        self.reason = Some(format!("failed to set up invariant testing environment: {e}"));
544        self

546
547    /// Returns the invariant test result.
548    pub fn invariant_result(
549        mut self,
550        gas_report_traces: Vec<Vec<CallTraceArena>>,
551        success: bool,
552        reason: Option<String>,
553        counterexample: Option<CounterExample>,
554        cases: Vec<FuzzedCases>,
555        reverts: usize,
556    ) -> Self {
557        self.kind = TestKind::Invariant {
558            runs: cases.len(),
559            calls: cases.iter().map(|sequence| sequence.cases().len()).sum(),
560            reverts,
561        };
562        self.status = match success {
563            true => TestStatus::Success,
564            false => TestStatus::Failure,
565        };
566        self.reason = reason;
567        self.counterexample = counterexample;
568        self.gas_report_traces = gas_report_traces;
569        self
570    }

572    /// Returns `true` if this is the result of a fuzz test
573    pub fn is_fuzz(&self) -> bool {
574        matches!(self.kind, TestKind::Fuzz { .. })
575    }

577    /// Formats the test result into a string (for printing).
578    pub fn short_result(&self, name: &str) -> String {
579        format!("{self} {name} {}", self.kind.report())
580    }
581
582    /// Function to merge logs, addresses, traces and coverage from a call result into test result.
583    pub fn merge_call_result(&mut self, call_result: &RawCallResult) {
584        self.logs.extend(call_result.logs.clone());
585        self.labeled_addresses.extend(call_result.labels.clone());
586        self.traces.extend(call_result.traces.clone().map(|traces| (TraceKind::Execution, traces)));
587        self.merge_coverages(call_result.coverage.clone());
588    }
589
590    /// Function to merge given coverage in current test result coverage.
591    pub fn merge_coverages(&mut self, other_coverage: Option<HitMaps>) {
592        let old_coverage = std::mem::take(&mut self.coverage);
593        self.coverage = match (old_coverage, other_coverage) {
594            (Some(old_coverage), Some(other)) => Some(old_coverage.merged(other)),
595            (a, b) => a.or(b),
596        };
597    }

600/// Data report by a test.
601#[derive(Clone, Debug, PartialEq, Eq)]
602pub enum TestKindReport {
603    Unit { gas: u64 },
604    Fuzz { runs: usize, mean_gas: u64, median_gas: u64 },
605    Invariant { runs: usize, calls: usize, reverts: usize },
606}
607
608impl fmt::Display for TestKindReport {
609    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
610        match *self {
611            Self::Unit { gas } => {
612                write!(f, "(gas: {gas})")
613            }
614            Self::Fuzz { runs, mean_gas, median_gas } => {
615                write!(f, "(runs: {runs}, : {mean_gas}, ~: {median_gas})")
616            }
617            Self::Invariant { runs, calls, reverts } => {
618                write!(f, "(runs: {runs}, calls: {calls}, reverts: {reverts})")
619            }
620        }
621    }
622}
623
624impl TestKindReport {
625    /// Returns the main gas value to compare against
626    pub fn gas(&self) -> u64 {
627        match *self {
628            Self::Unit { gas } => gas,
629            // We use the median for comparisons
630            Self::Fuzz { median_gas, .. } => median_gas,
631            // We return 0 since it's not applicable
632            Self::Invariant { .. } => 0,
633        }
634    }
635}

637/// Various types of tests
638#[derive(Clone, Debug, Serialize, Deserialize)]
639pub enum TestKind {
640    /// A unit test.
641    Unit { gas: u64 },
642    /// A fuzz test.
643    Fuzz {
644        /// we keep this for the debugger
645        first_case: FuzzCase,
646        runs: usize,
647        mean_gas: u64,
648        median_gas: u64,
649    },
650    /// An invariant test.
651    Invariant { runs: usize, calls: usize, reverts: usize },
652}
653
654impl Default for TestKind {
655    fn default() -> Self {
656        Self::Unit { gas: 0 }
657    }
658}
659
660impl TestKind {
661    /// The gas consumed by this test
662    pub fn report(&self) -> TestKindReport {
663        match *self {
664            Self::Unit { gas } => TestKindReport::Unit { gas },
665            Self::Fuzz { first_case: _, runs, mean_gas, median_gas } => {
666                TestKindReport::Fuzz { runs, mean_gas, median_gas }
667            }
668            Self::Invariant { runs, calls, reverts } => {
669                TestKindReport::Invariant { runs, calls, reverts }
670            }
671        }
672    }
673}

693impl TestSetup {
694    pub fn from_evm_error_with(
695        error: EvmError,
696        mut logs: Vec<Log>,
697        mut traces: Traces,
698        mut labeled_addresses: HashMap<Address, String>,
699    ) -> Self {
700        match error {
701            EvmError::Execution(err) => {
702                // force the tracekind to be setup so a trace is shown.
703                traces.extend(err.raw.traces.map(|traces| (TraceKind::Setup, traces)));
704                logs.extend(err.raw.logs);
705                labeled_addresses.extend(err.raw.labels);
706                Self::failed_with(logs, traces, labeled_addresses, err.reason)
707            }
708            e => Self::failed_with(
709                logs,
710                traces,
711                labeled_addresses,
712                format!("failed to deploy contract: {e}"),
713            ),
714        }
715    }
716
717    pub fn success(
718        address: Address,
719        logs: Vec<Log>,
720        traces: Traces,
721        labeled_addresses: HashMap<Address, String>,
722        coverage: Option<HitMaps>,
723        fuzz_fixtures: FuzzFixtures,
724    ) -> Self {
725        Self { address, logs, traces, labeled_addresses, reason: None, coverage, fuzz_fixtures }
726    }

728    pub fn failed_with(
729        logs: Vec<Log>,
730        traces: Traces,
731        labeled_addresses: HashMap<Address, String>,
732        reason: String,
733    ) -> Self {
734        Self {
735            address: Address::ZERO,
736            logs,
737            traces,
738            labeled_addresses,
739            reason: Some(reason),
740            coverage: None,
741            fuzz_fixtures: FuzzFixtures::default(),
742        }
743    }
744

746        Self { reason: Some(reason), ..Default::default() }
747    }
