11/// length of the name column for pretty formatting `{:>20}{value}`
12const NAME_COLUMN_LEN: usize = 20usize;
13
14/// Helper trait to format Ethereum types.
15///
16/// # Examples
17///
18/// ```
19/// use foundry_common_fmt::UIfmt;
20///
21/// let boolean: bool = true;
22/// let string = boolean.pretty();
23/// ```
24pub trait UIfmt {
25    /// Return a prettified string version of the value
26    fn pretty(&self) -> String;
27}
28
29impl<T: UIfmt> UIfmt for &T {
30    fn pretty(&self) -> String {
31        (*self).pretty()
32    }
33}
34
35impl<T: UIfmt> UIfmt for Option<T> {
36    fn pretty(&self) -> String {
37        if let Some(ref inner) = self {
38            inner.pretty()
39        } else {
40            String::new()
41        }
42    }
43}
44
45impl<T: UIfmt> UIfmt for [T] {
46    fn pretty(&self) -> String {
47        if !self.is_empty() {
48            let mut s = String::with_capacity(self.len() * 64);
49            s.push_str("[\n");
50            for item in self {
51                for line in item.pretty().lines() {
52                    s.push('\t');
53                    s.push_str(line);
54                    s.push('\n');
55                }
56            }
57            s.push(']');
58            s
59        } else {
60            "[]".to_string()
61        }
62    }
63}

158impl UIfmt for AnyTransactionReceipt {
159    fn pretty(&self) -> String {
160        let Self {
161            inner:
162                TransactionReceipt {
163                    transaction_hash,
164                    transaction_index,
165                    block_hash,
166                    block_number,
167                    from,
168                    to,
169                    gas_used,
170                    contract_address,
171                    state_root,
172                    effective_gas_price,
173                    inner:
174                        AnyReceiptEnvelope {
175                            r#type: transaction_type,
176                            inner:
177                                ReceiptWithBloom {
178                                    receipt: Receipt { status, cumulative_gas_used, logs },
179                                    logs_bloom,
180                                },
181                        },
182                    blob_gas_price,
183                    blob_gas_used,
184                    authorization_list,
185                },
186            other,
187        } = self;
188
189        let mut pretty = format!(
190            "
191blockHash               {}
192blockNumber             {}
193contractAddress         {}
194cumulativeGasUsed       {}
195effectiveGasPrice       {}
196from                    {}
197gasUsed                 {}
198logs                    {}
199logsBloom               {}
200root                    {}
201status                  {}
202transactionHash         {}
203transactionIndex        {}
204type                    {}
205blobGasPrice            {}
206blobGasUsed             {}
207authorizationList       {}",
208            block_hash.pretty(),
209            block_number.pretty(),
210            contract_address.pretty(),
211            cumulative_gas_used.pretty(),
212            effective_gas_price.pretty(),
213            from.pretty(),
214            gas_used.pretty(),
215            serde_json::to_string(&logs).unwrap(),
216            logs_bloom.pretty(),
217            state_root.pretty(),
218            status.pretty(),
219            transaction_hash.pretty(),
220            transaction_index.pretty(),
221            transaction_type,
222            blob_gas_price.pretty(),
223            blob_gas_used.pretty(),
224            authorization_list
225                .as_ref()
226                .map(|l| serde_json::to_string(&l).unwrap())
227                .unwrap_or_default(),
228        );
229
230        if let Some(to) = to {
231            pretty.push_str(&format!("\nto                      {}", to.pretty()));
232        }
233
234        // additional captured fields
235        for (key, val) in other.iter() {
236            pretty.push_str(&format!("\n{key}             {val}"));
237        }
238
239        pretty
240    }
241}

291impl UIfmt for OtherFields {
292    fn pretty(&self) -> String {
293        let mut s = String::with_capacity(self.len() * 30);
294        if !self.is_empty() {
295            s.push('\n');
296        }
297        for (key, value) in self.iter() {
298            let val = EthValue::from(value.clone()).pretty();
299            let offset = NAME_COLUMN_LEN.saturating_sub(key.len());
300            s.push_str(key);
301            s.extend(std::iter::repeat(' ').take(offset + 1));
302            s.push_str(&val);
303            s.push('\n');
304        }
305        s
306    }
307}

346/// Various numerical ethereum types used for pretty printing
347#[derive(Clone, Debug, Deserialize)]
348#[serde(untagged)]
349#[allow(missing_docs)]
350pub enum EthValue {
351    U64(U64),
352    U256(U256),
353    U64Array(Vec<U64>),
354    U256Array(Vec<U256>),
355    Other(serde_json::Value),
356}
357
358impl From<serde_json::Value> for EthValue {
359    fn from(val: serde_json::Value) -> Self {
360        serde_json::from_value(val).expect("infallible")
361    }
362}
363
364impl UIfmt for EthValue {
365    fn pretty(&self) -> String {
366        match self {
367            Self::U64(num) => num.pretty(),
368            Self::U256(num) => num.pretty(),
369            Self::U64Array(arr) => arr.pretty(),
370            Self::U256Array(arr) => arr.pretty(),
371            Self::Other(val) => val.to_string().trim_matches('"').to_string(),
372        }
373    }
374}

376/// Returns the `UiFmt::pretty()` formatted attribute of the transactions
377pub fn get_pretty_tx_attr(transaction: &Transaction, attr: &str) -> Option<String> {
378    match attr {
379        "blockHash" | "block_hash" => Some(transaction.block_hash.pretty()),
380        "blockNumber" | "block_number" => Some(transaction.block_number.pretty()),
381        "from" => Some(transaction.from.pretty()),
382        "gas" => Some(transaction.gas.pretty()),
383        "gasPrice" | "gas_price" => Some(transaction.gas_price.pretty()),
384        "hash" => Some(transaction.hash.pretty()),
385        "input" => Some(transaction.input.pretty()),
386        "nonce" => Some(transaction.nonce.to_string()),
387        "s" => transaction.signature.map(|s| B256::from(s.s).pretty()),
388        "r" => transaction.signature.map(|s| B256::from(s.r).pretty()),
389        "to" => Some(transaction.to.pretty()),
390        "transactionIndex" | "transaction_index" => Some(transaction.transaction_index.pretty()),
391        "v" => transaction.signature.map(|s| s.v.pretty()),
392        "value" => Some(transaction.value.pretty()),
393        other => {
394            if let Some(value) = transaction.other.get(other) {
395                return Some(value.to_string().trim_matches('"').to_string())
396            }
397            None
398        }
399    }
400}

402/// Returns the `UiFmt::pretty()` formatted attribute of the given block
403pub fn get_pretty_block_attr(block: &Block, attr: &str) -> Option<String> {
404    match attr {
405        "baseFeePerGas" | "base_fee_per_gas" => Some(block.header.base_fee_per_gas.pretty()),
406        "difficulty" => Some(block.header.difficulty.pretty()),
407        "extraData" | "extra_data" => Some(block.header.extra_data.pretty()),
408        "gasLimit" | "gas_limit" => Some(block.header.gas_limit.pretty()),
409        "gasUsed" | "gas_used" => Some(block.header.gas_used.pretty()),
410        "hash" => Some(block.header.hash.pretty()),
411        "logsBloom" | "logs_bloom" => Some(block.header.logs_bloom.pretty()),
412        "miner" | "author" => Some(block.header.miner.pretty()),
413        "mixHash" | "mix_hash" => Some(block.header.mix_hash.pretty()),
414        "nonce" => Some(block.header.nonce.pretty()),
415        "number" => Some(block.header.number.pretty()),
416        "parentHash" | "parent_hash" => Some(block.header.parent_hash.pretty()),
417        "transactionsRoot" | "transactions_root" => Some(block.header.transactions_root.pretty()),
418        "receiptsRoot" | "receipts_root" => Some(block.header.receipts_root.pretty()),
419        "sha3Uncles" | "sha_3_uncles" => Some(block.header.uncles_hash.pretty()),
420        "size" => Some(block.size.pretty()),
421        "stateRoot" | "state_root" => Some(block.header.state_root.pretty()),
422        "timestamp" => Some(block.header.timestamp.pretty()),
423        "totalDifficulty" | "total_difficult" => Some(block.header.total_difficulty.pretty()),
424        other => {
425            if let Some(value) = block.other.get(other) {
426                let val = EthValue::from(value.clone());
427                return Some(val.pretty())
428            }
429            None
430        }
431    }
432}
