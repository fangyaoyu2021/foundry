75/// Creates the node and runs the server.
76///
77/// Returns the [EthApi] that can be used to interact with the node and the [JoinHandle] of the
78/// task.
79///
80/// # Panics
81///
82/// Panics if any error occurs. For a non-panicking version, use [`try_spawn`].
83///
84///
85/// # Examples
86///
87/// ```no_run
88/// # use anvil::NodeConfig;
89/// # async fn spawn() -> eyre::Result<()> {
90/// let config = NodeConfig::default();
91/// let (api, handle) = anvil::spawn(config).await;
92///
93/// // use api
94///
95/// // wait forever
96/// handle.await.unwrap().unwrap();
97/// # Ok(())
98/// # }
99/// ```
100pub async fn spawn(config: NodeConfig) -> (EthApi, NodeHandle) {
101    try_spawn(config).await.expect("failed to spawn node")
102}

104/// Creates the node and runs the server
105///
106/// Returns the [EthApi] that can be used to interact with the node and the [JoinHandle] of the
107/// task.
108///
109/// # Examples
110///
111/// ```no_run
112/// # use anvil::NodeConfig;
113/// # async fn spawn() -> eyre::Result<()> {
114/// let config = NodeConfig::default();
115/// let (api, handle) = anvil::try_spawn(config).await?;
116///
117/// // use api
118///
119/// // wait forever
120/// handle.await??;
121/// # Ok(())
122/// # }
123/// ```
124pub async fn try_spawn(mut config: NodeConfig) -> io::Result<(EthApi, NodeHandle)> {
125    let logger = if config.enable_tracing { init_tracing() } else { Default::default() };
126    logger.set_enabled(!config.silent);
127
128    let backend = Arc::new(config.setup().await);
129
130    if config.enable_auto_impersonate {
131        backend.auto_impersonate_account(true);
132    }
133
134    let fork = backend.get_fork();
135
136    let NodeConfig {
137        signer_accounts,
138        block_time,
139        port,
140        max_transactions,
141        server_config,
142        no_mining,
143        transaction_order,
144        genesis,
145        mixed_mining,
146        ..
147    } = config.clone();
148
149    let pool = Arc::new(Pool::default());
150
151    let mode = if let Some(block_time) = block_time {
152        if mixed_mining {
153            let listener = pool.add_ready_listener();
154            MiningMode::mixed(max_transactions, listener, block_time)
155        } else {
156            MiningMode::interval(block_time)
157        }
158    } else if no_mining {
159        MiningMode::None
160    } else {
161        // get a listener for ready transactions
162        let listener = pool.add_ready_listener();
163        MiningMode::instant(max_transactions, listener)
164    };
165
166    let miner = match &fork {
167        Some(fork) => {
168            Miner::new(mode).with_forced_transactions(fork.config.read().force_transactions.clone())
169        }
170        _ => Miner::new(mode),
171    };
172
173    let dev_signer: Box<dyn EthSigner> = Box::new(DevSigner::new(signer_accounts));
174    let mut signers = vec![dev_signer];
175    if let Some(genesis) = genesis {
176        let genesis_signers = genesis
177            .alloc
178            .values()
179            .filter_map(|acc| acc.private_key)
180            .flat_map(|k| PrivateKeySigner::from_bytes(&k))
181            .collect::<Vec<_>>();
182        if !genesis_signers.is_empty() {
183            signers.push(Box::new(DevSigner::new(genesis_signers)));
184        }
185    }
186
187    let fee_history_cache = Arc::new(Mutex::new(Default::default()));
188    let fee_history_service = FeeHistoryService::new(
189        backend.new_block_notifications(),
190        Arc::clone(&fee_history_cache),
191        StorageInfo::new(Arc::clone(&backend)),
192    );
193    // create an entry for the best block
194    if let Some(header) = backend.get_block(backend.best_number()).map(|block| block.header) {
195        fee_history_service.insert_cache_entry_for_block(header.hash_slow(), &header);
196    }
197
198    let filters = Filters::default();
199
200    // create the cloneable api wrapper
201    let api = EthApi::new(
202        Arc::clone(&pool),
203        Arc::clone(&backend),
204        Arc::new(signers),
205        fee_history_cache,
206        fee_history_service.fee_history_limit(),
207        miner.clone(),
208        logger,
209        filters.clone(),
210        transaction_order,
211    );
212
213    // spawn the node service
214    let node_service =
215        tokio::task::spawn(NodeService::new(pool, backend, miner, fee_history_service, filters));
216
217    let mut servers = Vec::with_capacity(config.host.len());
218    let mut addresses = Vec::with_capacity(config.host.len());
219
220    for addr in &config.host {
221        let sock_addr = SocketAddr::new(*addr, port);
222
223        // Create a TCP listener.
224        let tcp_listener = tokio::net::TcpListener::bind(sock_addr).await?;
225        addresses.push(tcp_listener.local_addr()?);
226
227        // Spawn the server future on a new task.
228        let srv = server::serve_on(tcp_listener, api.clone(), server_config.clone());
229        servers.push(tokio::task::spawn(srv.map_err(Into::into)));
230    }
231
232    let tokio_handle = Handle::current();
233    let (signal, on_shutdown) = shutdown::signal();
234    let task_manager = TaskManager::new(tokio_handle, on_shutdown);
235
236    let ipc_task =
237        config.get_ipc_path().map(|path| try_spawn_ipc(api.clone(), path)).transpose()?;
238
239    let handle = NodeHandle {
240        config,
241        node_service,
242        servers,
243        ipc_task,
244        addresses,
245        _signal: Some(signal),
246        task_manager,
247    };
248
249    handle.print(fork.as_ref());
250
251    Ok((api, handle))
252}

256/// A handle to the spawned node and server tasks
257///
258/// This future will resolve if either the node or server task resolve/fail.
259pub struct NodeHandle {
260    config: NodeConfig,
261    /// The address of the running rpc server
262    addresses: Vec<SocketAddr>,
263    /// Join handle for the Node Service
264    pub node_service: JoinHandle<Result<(), NodeError>>,
265    /// Join handles (one per socket) for the Anvil server.
266    pub servers: Vec<JoinHandle<Result<(), NodeError>>>,
267    // The future that joins the ipc server, if any
268    ipc_task: Option<IpcTask>,
269    /// A signal that fires the shutdown, fired on drop.
270    _signal: Option<Signal>,
271    /// A task manager that can be used to spawn additional tasks
272    task_manager: TaskManager,
273}

275impl NodeHandle {
276    /// The [NodeConfig] the node was launched with

278        &self.config
279    }
280
281    /// Prints the launch info
282    pub(crate) fn print(&self, fork: Option<&ClientFork>) {
283        self.config.print(fork);
284        if !self.config.silent {
285            if let Some(ipc_path) = self.ipc_path() {
286                println!("IPC path: {ipc_path}");
287            }
288            println!(
289                "Listening on {}",
290                self.addresses
291                    .iter()
292                    .map(|addr| { addr.to_string() })
293                    .collect::<Vec<String>>()
294                    .join(", ")
295            );
296        }
297    }
298
299    /// The address of the launched server
300    ///
301    /// **N.B.** this may not necessarily be the same `host + port` as configured in the
302    /// `NodeConfig`, if port was set to 0, then the OS auto picks an available port
303    pub fn socket_address(&self) -> &SocketAddr {
304        &self.addresses[0]
305    }
306
307    /// Returns the http endpoint
308    pub fn http_endpoint(&self) -> String {
309        format!("http://{}", self.socket_address())
310    }
311
312    /// Returns the websocket endpoint

314        format!("ws://{}", self.socket_address())
315    }
316
317    /// Returns the path of the launched ipc server, if any
318    pub fn ipc_path(&self) -> Option<String> {
319        self.config.get_ipc_path()
320    }
321
322    /// Constructs a [`RetryProvider`] for this handle's HTTP endpoint.
323    pub fn http_provider(&self) -> RetryProvider {
324        ProviderBuilder::new(&self.http_endpoint()).build().expect("failed to build HTTP provider")
325    }
326
327    /// Constructs a [`RetryProvider`] for this handle's WS endpoint.
328    pub fn ws_provider(&self) -> RetryProvider {
329        ProviderBuilder::new(&self.ws_endpoint()).build().expect("failed to build WS provider")
330    }
331
332    /// Constructs a [`RetryProvider`] for this handle's IPC endpoint, if any.
333    pub fn ipc_provider(&self) -> Option<RetryProvider> {
334        ProviderBuilder::new(&self.config.get_ipc_path()?).build().ok()
335    }
336
337    /// Signer accounts that can sign messages/transactions from the EVM node
338    pub fn dev_accounts(&self) -> impl Iterator<Item = Address> + '_ {
339        self.config.signer_accounts.iter().map(|wallet| wallet.address())
340    }
341
342    /// Signer accounts that can sign messages/transactions from the EVM node
343    pub fn dev_wallets(&self) -> impl Iterator<Item = PrivateKeySigner> + '_ {
344        self.config.signer_accounts.iter().cloned()
345    }
346
347    /// Accounts that will be initialised with `genesis_balance` in the genesis block
348    pub fn genesis_accounts(&self) -> impl Iterator<Item = Address> + '_ {
349        self.config.genesis_accounts.iter().map(|w| w.address())
350    }
351
352    /// Native token balance of every genesis account in the genesis block
353    pub fn genesis_balance(&self) -> U256 {
354        self.config.genesis_balance
355    }
356
357    /// Default gas price for all txs
358    pub fn gas_price(&self) -> u128 {
359        self.config.get_gas_price()
360    }
361
362    /// Returns the shutdown signal
363    pub fn shutdown_signal(&self) -> &Option<Signal> {
364        &self._signal
365    }
366
367    /// Returns mutable access to the shutdown signal
368    ///
369    /// This can be used to extract the Signal
370    pub fn shutdown_signal_mut(&mut self) -> &mut Option<Signal> {
371        &mut self._signal
372    }

374    /// Returns the task manager that can be used to spawn new tasks
375    ///
376    /// ```
377    /// use anvil::NodeHandle;
378    /// # fn t(handle: NodeHandle) {
379    /// let task_manager = handle.task_manager();

394impl Future for NodeHandle {
395    type Output = Result<NodeResult<()>, JoinError>;
396
397    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
398        let pin = self.get_mut();
399
400        // poll the ipc task
401        if let Some(mut ipc) = pin.ipc_task.take() {
402            if let Poll::Ready(res) = ipc.poll_unpin(cx) {
403                return Poll::Ready(res.map(|()| Ok(())));
404            } else {
405                pin.ipc_task = Some(ipc);
406            }
407        }
408
409        // poll the node service task
410        if let Poll::Ready(res) = pin.node_service.poll_unpin(cx) {
411            return Poll::Ready(res);
412        }
413
414        // poll the axum server handles
415        for server in pin.servers.iter_mut() {
416            if let Poll::Ready(res) = server.poll_unpin(cx) {
417                return Poll::Ready(res);
418            }
419        }
420
421        Poll::Pending
422    }
423}

426pub fn init_tracing() -> LoggingManager {
427    use tracing_subscriber::prelude::*;
428
429    let manager = LoggingManager::default();
430    // check whether `RUST_LOG` is explicitly set
431    let _ = if std::env::var("RUST_LOG").is_ok() {
432        tracing_subscriber::Registry::default()
433            .with(tracing_subscriber::EnvFilter::from_default_env())
434            .with(tracing_subscriber::fmt::layer())
435            .try_init()
436    } else {
437        tracing_subscriber::Registry::default()
438            .with(NodeLogLayer::new(manager.clone()))
439            .with(
440                tracing_subscriber::fmt::layer()
441                    .without_time()
442                    .with_target(false)
443                    .with_level(false),
444            )
445            .try_init()
446    };
447
448    manager
