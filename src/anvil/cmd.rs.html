199    pub fn into_node_config(self) -> NodeConfig {
200        let genesis_balance = Unit::ETHER.wei().saturating_mul(U256::from(self.balance));
201        let compute_units_per_second = if self.evm_opts.no_rate_limit {
202            Some(u64::MAX)
203        } else {
204            self.evm_opts.compute_units_per_second
205        };
206
207        NodeConfig::default()
208            .with_gas_limit(self.evm_opts.gas_limit)
209            .disable_block_gas_limit(self.evm_opts.disable_block_gas_limit)
210            .with_gas_price(self.evm_opts.gas_price)
211            .with_hardfork(self.hardfork)
212            .with_blocktime(self.block_time)
213            .with_no_mining(self.no_mining)
214            .with_mixed_mining(self.mixed_mining, self.block_time)
215            .with_account_generator(self.account_generator())
216            .with_genesis_balance(genesis_balance)
217            .with_genesis_timestamp(self.timestamp)
218            .with_port(self.port)
219            .with_fork_choice(
220                match (self.evm_opts.fork_block_number, self.evm_opts.fork_transaction_hash) {
221                    (Some(block), None) => Some(ForkChoice::Block(block)),
222                    (None, Some(hash)) => Some(ForkChoice::Transaction(hash)),
223                    _ => {
224                        self.evm_opts.fork_url.as_ref().and_then(|f| f.block).map(ForkChoice::Block)
225                    }
226                },
227            )
228            .with_fork_headers(self.evm_opts.fork_headers)
229            .with_fork_chain_id(self.evm_opts.fork_chain_id.map(u64::from).map(U256::from))
230            .fork_request_timeout(self.evm_opts.fork_request_timeout.map(Duration::from_millis))
231            .fork_request_retries(self.evm_opts.fork_request_retries)
232            .fork_retry_backoff(self.evm_opts.fork_retry_backoff.map(Duration::from_millis))
233            .fork_compute_units_per_second(compute_units_per_second)
234            .with_eth_rpc_url(self.evm_opts.fork_url.map(|fork| fork.url))
235            .with_base_fee(self.evm_opts.block_base_fee_per_gas)
236            .with_storage_caching(self.evm_opts.no_storage_caching)
237            .with_server_config(self.server_config)
238            .with_host(self.host)
239            .set_silent(self.silent)
240            .set_config_out(self.config_out)
241            .with_chain_id(self.evm_opts.chain_id)
242            .with_transaction_order(self.order)
243            .with_genesis(self.init)
244            .with_steps_tracing(self.evm_opts.steps_tracing)
245            .with_print_logs(!self.evm_opts.disable_console_log)
246            .with_auto_impersonate(self.evm_opts.auto_impersonate)
247            .with_ipc(self.ipc)
248            .with_code_size_limit(self.evm_opts.code_size_limit)
249            .set_pruned_history(self.prune_history)
250            .with_init_state(self.load_state.or_else(|| self.state.and_then(|s| s.state)))
251            .with_transaction_block_keeper(self.transaction_block_keeper)

259    fn account_generator(&self) -> AccountGenerator {
260        let mut gen = AccountGenerator::new(self.accounts as usize)
261            .phrase(DEFAULT_MNEMONIC)
262            .chain_id(self.evm_opts.chain_id.unwrap_or_else(|| CHAIN_ID.into()));
263        if let Some(ref mnemonic) = self.mnemonic {
264            gen = gen.phrase(mnemonic);
265        } else if let Some(count) = self.mnemonic_random {
266            let mut rng = rand::thread_rng();
267            let mnemonic = match Mnemonic::<English>::new_with_count(&mut rng, count) {
268                Ok(mnemonic) => mnemonic.to_phrase(),
269                Err(_) => DEFAULT_MNEMONIC.to_string(),
270            };
271            gen = gen.phrase(mnemonic);
272        } else if let Some(seed) = self.mnemonic_seed {
273            let mut seed = StdRng::seed_from_u64(seed);
274            let mnemonic = Mnemonic::<English>::new(&mut seed).to_phrase();
275            gen = gen.phrase(mnemonic);
276        }
277        if let Some(ref derivation) = self.derivation_path {
278            gen = gen.derivation_path(derivation);
279        }
280        gen
281    }

288    /// Starts the node
289    ///
290    /// See also [crate::spawn()]
291    pub async fn run(self) -> eyre::Result<()> {
292        let dump_state = self.dump_state_path();
293        let dump_interval =
294            self.state_interval.map(Duration::from_secs).unwrap_or(DEFAULT_DUMP_INTERVAL);
295
296        let (api, mut handle) = crate::try_spawn(self.into_node_config()).await?;
297
298        // sets the signal handler to gracefully shutdown.
299        let mut fork = api.get_fork();
300        let running = Arc::new(AtomicUsize::new(0));
301
302        // handle for the currently running rt, this must be obtained before setting the crtlc
303        // handler, See [Handle::current]
304        let mut signal = handle.shutdown_signal_mut().take();
305
306        let task_manager = handle.task_manager();
307        let mut on_shutdown = task_manager.on_shutdown();
308
309        let mut state_dumper = PeriodicStateDumper::new(api, dump_state, dump_interval);
310
311        task_manager.spawn(async move {
312            // wait for the SIGTERM signal on unix systems
313            #[cfg(unix)]
314            let mut sigterm = Box::pin(async {
315                if let Ok(mut stream) =
316                    tokio::signal::unix::signal(tokio::signal::unix::SignalKind::terminate())
317                {
318                    stream.recv().await;
319                } else {
320                    futures::future::pending::<()>().await;
321                }
322            });
323
324            // on windows, this will never fires
325            #[cfg(not(unix))]
326            let mut sigterm = Box::pin(futures::future::pending::<()>());
327
328            // await shutdown signal but also periodically flush state
329            tokio::select! {
330                 _ = &mut sigterm => {
331                    trace!("received sigterm signal, shutting down");
332                },
333                _ = &mut on_shutdown =>{
334
335                }
336                _ = &mut state_dumper =>{}
337            }
338
339            // shutdown received
340            state_dumper.dump().await;
341
342            // cleaning up and shutting down
343            // this will make sure that the fork RPC cache is flushed if caching is configured
344            if let Some(fork) = fork.take() {
345                trace!("flushing cache on shutdown");
346                fork.database
347                    .read()
348                    .await
349                    .maybe_flush_cache()
350                    .expect("Could not flush cache on fork DB");
351                // cleaning up and shutting down
352                // this will make sure that the fork RPC cache is flushed if caching is configured
353            }
354            std::process::exit(0);
355        });
356
357        ctrlc::set_handler(move || {
358            let prev = running.fetch_add(1, Ordering::SeqCst);
359            if prev == 0 {
360                trace!("received shutdown signal, shutting down");
361                let _ = signal.take();
362            }
363        })
364        .expect("Error setting Ctrl-C handler");
365
366        Ok(handle.await??)
367    }

559/// Helper type to periodically dump the state of the chain to disk
560struct PeriodicStateDumper {
561    in_progress_dump: Option<Pin<Box<dyn Future<Output = ()> + Send + Sync + 'static>>>,
562    api: EthApi,
563    dump_state: Option<PathBuf>,
564    interval: Interval,
565}
566
567impl PeriodicStateDumper {
568    fn new(api: EthApi, dump_state: Option<PathBuf>, interval: Duration) -> Self {
569        let dump_state = dump_state.map(|mut dump_state| {
570            if dump_state.is_dir() {
571                dump_state = dump_state.join("state.json");
572            }
573            dump_state
574        });
575
576        // periodically flush the state
577        let interval = tokio::time::interval_at(Instant::now() + interval, interval);
578        Self { in_progress_dump: None, api, dump_state, interval }
579    }
580
581    async fn dump(&self) {
582        if let Some(state) = self.dump_state.clone() {
583            Self::dump_state(self.api.clone(), state).await
584        }
585    }
586
587    /// Infallible state dump
588    async fn dump_state(api: EthApi, dump_state: PathBuf) {
589        trace!(path=?dump_state, "Dumping state on shutdown");
590        match api.serialized_state().await {
591            Ok(state) => {
592                if let Err(err) = foundry_common::fs::write_json_file(&dump_state, &state) {
593                    error!(?err, "Failed to dump state");
594                } else {
595                    trace!(path=?dump_state, "Dumped state on shutdown");
596                }
597            }
598            Err(err) => {
599                error!(?err, "Failed to extract state");
600            }
601        }
602    }
603}
604
605// An endless future that periodically dumps the state to disk if configured.
606impl Future for PeriodicStateDumper {
607    type Output = ();
608
609    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
610        let this = self.get_mut();
611        if this.dump_state.is_none() {
612            return Poll::Pending
613        }
614
615        loop {
616            if let Some(mut flush) = this.in_progress_dump.take() {
617                match flush.poll_unpin(cx) {
618                    Poll::Ready(_) => {
619                        this.interval.reset();
620                    }
621                    Poll::Pending => {
622                        this.in_progress_dump = Some(flush);
623                        return Poll::Pending
624                    }
625                }
626            }
627
628            if this.interval.poll_tick(cx).is_ready() {
629                let api = this.api.clone();
630                let path = this.dump_state.clone().expect("exists; see above");
631                this.in_progress_dump = Some(Box::pin(Self::dump_state(api, path)));
632            } else {
633                break
634            }
635        }
636
637        Poll::Pending
638    }
639}

641/// Represents the --state flag and where to load from, or dump the state to
642#[derive(Clone, Debug)]
643pub struct StateFile {
644    pub path: PathBuf,
645    pub state: Option<SerializableState>,
646}
647
648impl StateFile {
649    /// This is used as the clap `value_parser` implementation to parse from file but only if it
650    /// exists
651    fn parse(path: &str) -> Result<Self, String> {
652        Self::parse_path(path)
653    }
654
655    /// Parse from file but only if it exists
656    pub fn parse_path(path: impl AsRef<Path>) -> Result<Self, String> {
657        let mut path = path.as_ref().to_path_buf();
658        if path.is_dir() {
659            path = path.join("state.json");
660        }
661        let mut state = Self { path, state: None };
662        if !state.path.exists() {
663            return Ok(state)
664        }
665
666        state.state = Some(SerializableState::load(&state.path).map_err(|err| err.to_string())?);
667
668        Ok(state)
669    }
670}

672/// Represents the input URL for a fork with an optional trailing block number:
673/// `http://localhost:8545@1000000`
674#[derive(Clone, Debug, PartialEq, Eq)]
675pub struct ForkUrl {
676    /// The endpoint url
677    pub url: String,
678    /// Optional trailing block
679    pub block: Option<u64>,
680}
681
682impl fmt::Display for ForkUrl {
683    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
684        self.url.fmt(f)?;
685        if let Some(block) = self.block {
686            write!(f, "@{block}")?;
687        }
688        Ok(())
689    }
690}
691
692impl FromStr for ForkUrl {
693    type Err = String;
694
695    fn from_str(s: &str) -> Result<Self, Self::Err> {
696        if let Some((url, block)) = s.rsplit_once('@') {
697            if block == "latest" {
698                return Ok(Self { url: url.to_string(), block: None })
699            }
700            // this will prevent false positives for auths `user:password@example.com`
701            if !block.is_empty() && !block.contains(':') && !block.contains('.') {
702                let block: u64 = block
703                    .parse()
704                    .map_err(|_| format!("Failed to parse block number: `{block}`"))?;
705                return Ok(Self { url: url.to_string(), block: Some(block) })
706            }
707        }
708        Ok(Self { url: s.to_string(), block: None })
709    }
710}

713fn read_genesis_file(path: &str) -> Result<Genesis, String> {

717fn duration_from_secs_f64(s: &str) -> Result<Duration, String> {
718    let s = s.parse::<f64>().map_err(|e| e.to_string())?;
719    if s == 0.0 {
720        return Err("Duration must be greater than 0".to_string());
721    }
722    Duration::try_from_secs_f64(s).map_err(|e| e.to_string())
723}
