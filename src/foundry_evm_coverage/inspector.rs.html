1use crate::{HitMap, HitMaps};
2use revm::{interpreter::Interpreter, Database, EvmContext, Inspector};
3
4#[derive(Clone, Debug, Default)]
5pub struct CoverageCollector {
6    /// Maps that track instruction hit data.
7    pub maps: HitMaps,
8}
9
10impl<DB: Database> Inspector<DB> for CoverageCollector {
11    #[inline]
12    fn initialize_interp(&mut self, interp: &mut Interpreter, _context: &mut EvmContext<DB>) {
13        let hash = interp.contract.hash.expect("Contract hash is None");
14        self.maps
15            .entry(hash)
16            .or_insert_with(|| HitMap::new(interp.contract.bytecode.original_bytes()));
17    }
18
19    #[inline]
20    fn step(&mut self, interp: &mut Interpreter, _context: &mut EvmContext<DB>) {
21        let hash = interp.contract.hash.expect("Contract hash is None");
22        self.maps.entry(hash).and_modify(|map| map.hit(interp.program_counter()));
23    }
24}
