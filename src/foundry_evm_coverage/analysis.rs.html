8/// A visitor that walks the AST of a single contract and finds coverage items.
9#[derive(Clone, Debug)]
10pub struct ContractVisitor<'a> {
11    /// The source ID of the contract.
12    source_id: usize,
13    /// The source code that contains the AST being walked.
14    source: &'a str,
15
16    /// The name of the contract being walked.
17    contract_name: &'a Arc<str>,
18
19    /// The current branch ID
20    branch_id: usize,
21    /// Stores the last line we put in the items collection to ensure we don't push duplicate lines
22    last_line: usize,
23
24    /// Coverage items
25    pub items: Vec<CoverageItem>,
26}
27
28impl<'a> ContractVisitor<'a> {
29    pub fn new(source_id: usize, source: &'a str, contract_name: &'a Arc<str>) -> Self {
30        Self { source_id, source, contract_name, branch_id: 0, last_line: 0, items: Vec::new() }
31    }
32
33    pub fn visit_contract(&mut self, node: &Node) -> eyre::Result<()> {
34        // Find all functions and walk their AST
35        for node in &node.nodes {
36            match node.node_type {
37                NodeType::FunctionDefinition => {
38                    self.visit_function_definition(node)?;
39                }
40                NodeType::ModifierDefinition => {
41                    self.visit_modifier_or_yul_fn_definition(node)?;
42                }
43                _ => {}
44            }
45        }
46        Ok(())
47    }
48
49    fn visit_function_definition(&mut self, node: &Node) -> eyre::Result<()> {
50        let name: String =
51            node.attribute("name").ok_or_else(|| eyre::eyre!("Function has no name"))?;
52
53        // TODO(onbjerg): Figure out why we cannot find anchors for the receive function
54        let kind: String =
55            node.attribute("kind").ok_or_else(|| eyre::eyre!("Function has no kind"))?;
56        if kind == "receive" {
57            return Ok(())
58        }
59
60        match &node.body {
61            Some(body) => {
62                self.push_item(CoverageItem {
63                    kind: CoverageItemKind::Function { name },
64                    loc: self.source_location_for(&node.src),
65                    hits: 0,
66                });
67                self.visit_block(body)
68            }
69            _ => Ok(()),
70        }
71    }
72
73    fn visit_modifier_or_yul_fn_definition(&mut self, node: &Node) -> eyre::Result<()> {
74        let name: String =
75            node.attribute("name").ok_or_else(|| eyre::eyre!("Modifier has no name"))?;
76
77        match &node.body {
78            Some(body) => {
79                self.push_item(CoverageItem {
80                    kind: CoverageItemKind::Function { name },
81                    loc: self.source_location_for(&node.src),
82                    hits: 0,
83                });
84                self.visit_block(body)
85            }
86            _ => Ok(()),
87        }
88    }
89
90    fn visit_block(&mut self, node: &Node) -> eyre::Result<()> {
91        let statements: Vec<Node> = node.attribute("statements").unwrap_or_default();
92
93        for statement in &statements {
94            self.visit_statement(statement)?;
95        }
96
97        Ok(())
98    }
99
100    fn visit_statement(&mut self, node: &Node) -> eyre::Result<()> {
101        match node.node_type {
102            // Blocks
103            NodeType::Block | NodeType::UncheckedBlock | NodeType::YulBlock => {
104                self.visit_block(node)
105            }
106            // Inline assembly block
107            NodeType::InlineAssembly => self.visit_block(
108                &node
109                    .attribute("AST")
110                    .ok_or_else(|| eyre::eyre!("inline assembly block with no AST attribute"))?,
111            ),
112            // Simple statements
113            NodeType::Break |
114            NodeType::Continue |
115            NodeType::EmitStatement |
116            NodeType::RevertStatement |
117            NodeType::YulAssignment |
118            NodeType::YulBreak |
119            NodeType::YulContinue |
120            NodeType::YulLeave |
121            NodeType::YulVariableDeclaration => {
122                self.push_item(CoverageItem {
123                    kind: CoverageItemKind::Statement,
124                    loc: self.source_location_for(&node.src),
125                    hits: 0,
126                });
127                Ok(())
128            }
129            // Skip placeholder statements as they are never referenced in source maps.
130            NodeType::PlaceholderStatement => Ok(()),
131            // Return with eventual subcall
132            NodeType::Return => {
133                self.push_item(CoverageItem {
134                    kind: CoverageItemKind::Statement,
135                    loc: self.source_location_for(&node.src),
136                    hits: 0,
137                });
138                if let Some(expr) = node.attribute("expression") {
139                    self.visit_expression(&expr)?;
140                }
141                Ok(())
142            }
143            // Variable declaration
144            NodeType::VariableDeclarationStatement => {
145                self.push_item(CoverageItem {
146                    kind: CoverageItemKind::Statement,
147                    loc: self.source_location_for(&node.src),
148                    hits: 0,
149                });
150                if let Some(expr) = node.attribute("initialValue") {
151                    self.visit_expression(&expr)?;
152                }
153                Ok(())
154            }
155            // While loops
156            NodeType::DoWhileStatement | NodeType::WhileStatement => {
157                self.visit_expression(
158                    &node
159                        .attribute("condition")
160                        .ok_or_else(|| eyre::eyre!("while statement had no condition"))?,
161                )?;
162
163                let body = node
164                    .body
165                    .as_deref()
166                    .ok_or_else(|| eyre::eyre!("while statement had no body node"))?;
167                self.visit_block_or_statement(body)
168            }
169            // For loops
170            NodeType::ForStatement => {
171                if let Some(stmt) = node.attribute("initializationExpression") {
172                    self.visit_statement(&stmt)?;
173                }
174                if let Some(expr) = node.attribute("condition") {
175                    self.visit_expression(&expr)?;
176                }
177                if let Some(stmt) = node.attribute("loopExpression") {
178                    self.visit_statement(&stmt)?;
179                }
180
181                let body = node
182                    .body
183                    .as_deref()
184                    .ok_or_else(|| eyre::eyre!("for statement had no body node"))?;
185                self.visit_block_or_statement(body)
186            }
187            // Expression statement
188            NodeType::ExpressionStatement | NodeType::YulExpressionStatement => self
189                .visit_expression(
190                    &node
191                        .attribute("expression")
192                        .ok_or_else(|| eyre::eyre!("expression statement had no expression"))?,
193                ),
194            // If statement
195            NodeType::IfStatement => {
196                self.visit_expression(
197                    &node
198                        .attribute("condition")
199                        .ok_or_else(|| eyre::eyre!("if statement had no condition"))?,
200                )?;
201
202                let true_body: Node = node
203                    .attribute("trueBody")
204                    .ok_or_else(|| eyre::eyre!("if statement had no true body"))?;
205
206                // We need to store the current branch ID here since visiting the body of either of
207                // the if blocks may increase `self.branch_id` in the case of nested if statements.
208                let branch_id = self.branch_id;
209
210                // We increase the branch ID here such that nested branches do not use the same
211                // branch ID as we do.
212                self.branch_id += 1;
213
214                match node.attribute::<Node>("falseBody") {
215                    // Both if/else statements.
216                    Some(false_body) => {
217                        // Add branch coverage items only if one of true/branch bodies contains
218                        // statements.
219                        if has_statements(&true_body) || has_statements(&false_body) {
220                            // Add the coverage item for branch 0 (true body).
221                            // The relevant source range for the true branch is the `if(...)`
222                            // statement itself and the true body of the if statement.
223                            //
224                            // The false body of the statement is processed as its own thing.
225                            // If this source range is not processed like this, it is virtually
226                            // impossible to correctly map instructions back to branches that
227                            // include more complex logic like conditional logic.
228                            self.push_item(CoverageItem {
229                                kind: CoverageItemKind::Branch { branch_id, path_id: 0 },
230                                loc: self.source_location_for(&ast::LowFidelitySourceLocation {
231                                    start: node.src.start,
232                                    length: true_body.src.length.map(|length| {
233                                        true_body.src.start - node.src.start + length
234                                    }),
235                                    index: node.src.index,
236                                }),
237                                hits: 0,
238                            });
239                            // Add the coverage item for branch 1 (false body).
240                            // The relevant source range for the false branch is the `else`
241                            // statement itself and the false body of the else statement.
242                            self.push_item(CoverageItem {
243                                kind: CoverageItemKind::Branch { branch_id, path_id: 1 },
244                                loc: self.source_location_for(&ast::LowFidelitySourceLocation {
245                                    start: node.src.start,
246                                    length: false_body.src.length.map(|length| {
247                                        false_body.src.start - true_body.src.start + length
248                                    }),
249                                    index: node.src.index,
250                                }),
251                                hits: 0,
252                            });
253
254                            // Process the true body.
255                            self.visit_block_or_statement(&true_body)?;
256                            // Process the false body.
257                            self.visit_block_or_statement(&false_body)?;
258                        }
259                    }
260                    None => {
261                        // Add single branch coverage only if it contains statements.
262                        if has_statements(&true_body) {
263                            // Add the coverage item for branch 0 (true body).
264                            self.push_item(CoverageItem {
265                                kind: CoverageItemKind::SinglePathBranch { branch_id },
266                                loc: self.source_location_for(&true_body.src),
267                                hits: 0,
268                            });
269                            // Process the true body.
270                            self.visit_block_or_statement(&true_body)?;
271                        }
272                    }
273                }
274
275                Ok(())
276            }
277            NodeType::YulIf => {
278                self.visit_expression(
279                    &node
280                        .attribute("condition")
281                        .ok_or_else(|| eyre::eyre!("yul if statement had no condition"))?,
282                )?;
283                let body = node
284                    .body
285                    .as_deref()
286                    .ok_or_else(|| eyre::eyre!("yul if statement had no body"))?;
287
288                // We need to store the current branch ID here since visiting the body of either of
289                // the if blocks may increase `self.branch_id` in the case of nested if statements.
290                let branch_id = self.branch_id;
291
292                // We increase the branch ID here such that nested branches do not use the same
293                // branch ID as we do
294                self.branch_id += 1;
295
296                self.push_item(CoverageItem {
297                    kind: CoverageItemKind::Branch { branch_id, path_id: 0 },
298                    loc: self.source_location_for(&node.src),
299                    hits: 0,
300                });
301                self.visit_block(body)?;
302
303                Ok(())
304            }
305            // Try-catch statement. Coverage is reported for expression, for each clause and their
306            // bodies (if any).
307            NodeType::TryStatement => {
308                self.visit_expression(
309                    &node
310                        .attribute("externalCall")
311                        .ok_or_else(|| eyre::eyre!("try statement had no call"))?,
312                )?;
313
314                // Add coverage for each Try-catch clause.
315                for clause in node
316                    .attribute::<Vec<Node>>("clauses")
317                    .ok_or_else(|| eyre::eyre!("try statement had no clause"))?
318                {
319                    // Add coverage for clause statement.
320                    self.push_item(CoverageItem {
321                        kind: CoverageItemKind::Statement,
322                        loc: self.source_location_for(&clause.src),
323                        hits: 0,
324                    });
325                    self.visit_statement(&clause)?;
326
327                    // Add coverage for clause body only if it is not empty.
328                    if let Some(block) = clause.attribute::<Node>("block") {
329                        if has_statements(&block) {
330                            self.push_item(CoverageItem {
331                                kind: CoverageItemKind::Statement,
332                                loc: self.source_location_for(&block.src),
333                                hits: 0,
334                            });
335                            self.visit_block(&block)?;
336                        }
337                    }
338                }
339
340                Ok(())
341            }
342            NodeType::YulSwitch => {
343                // Add coverage for each case statement amd their bodies.
344                for case in node
345                    .attribute::<Vec<Node>>("cases")
346                    .ok_or_else(|| eyre::eyre!("yul switch had no case"))?
347                {
348                    self.push_item(CoverageItem {
349                        kind: CoverageItemKind::Statement,
350                        loc: self.source_location_for(&case.src),
351                        hits: 0,
352                    });
353                    self.visit_statement(&case)?;
354
355                    if let Some(body) = case.body {
356                        self.push_item(CoverageItem {
357                            kind: CoverageItemKind::Statement,
358                            loc: self.source_location_for(&body.src),
359                            hits: 0,
360                        });
361                        self.visit_block(&body)?
362                    }
363                }
364                Ok(())
365            }
366            NodeType::YulForLoop => {
367                if let Some(condition) = node.attribute("condition") {
368                    self.visit_expression(&condition)?;
369                }
370                if let Some(pre) = node.attribute::<Node>("pre") {
371                    self.visit_block(&pre)?
372                }
373                if let Some(post) = node.attribute::<Node>("post") {
374                    self.visit_block(&post)?
375                }
376
377                if let Some(body) = &node.body {
378                    self.push_item(CoverageItem {
379                        kind: CoverageItemKind::Statement,
380                        loc: self.source_location_for(&body.src),
381                        hits: 0,
382                    });
383                    self.visit_block(body)?
384                }
385                Ok(())
386            }
387            NodeType::YulFunctionDefinition => self.visit_modifier_or_yul_fn_definition(node),
388            _ => {
389                warn!("unexpected node type, expected a statement: {:?}", node.node_type);
390                Ok(())
391            }
392        }
393    }

395    fn visit_expression(&mut self, node: &Node) -> eyre::Result<()> {
396        match node.node_type {
397            NodeType::Assignment |
398            NodeType::UnaryOperation |
399            NodeType::Conditional |
400            NodeType::YulFunctionCall => {
401                self.push_item(CoverageItem {
402                    kind: CoverageItemKind::Statement,
403                    loc: self.source_location_for(&node.src),
404                    hits: 0,
405                });
406                Ok(())
407            }
408            NodeType::FunctionCall => {
409                // Do not count other kinds of calls towards coverage (like `typeConversion`
410                // and `structConstructorCall`).
411                let kind: Option<String> = node.attribute("kind");
412                if let Some("functionCall") = kind.as_deref() {
413                    self.push_item(CoverageItem {
414                        kind: CoverageItemKind::Statement,
415                        loc: self.source_location_for(&node.src),
416                        hits: 0,
417                    });
418
419                    let expr: Option<Node> = node.attribute("expression");
420                    if let Some(NodeType::Identifier) = expr.as_ref().map(|expr| &expr.node_type) {
421                        // Might be a require call, add branch coverage.
422                        let name: Option<String> = expr.and_then(|expr| expr.attribute("name"));
423                        if let Some("require" | "assert") = name.as_deref() {
424                            let branch_id = self.branch_id;
425                            self.branch_id += 1;
426                            self.push_item(CoverageItem {
427                                kind: CoverageItemKind::Branch { branch_id, path_id: 0 },
428                                loc: self.source_location_for(&node.src),
429                                hits: 0,
430                            });
431                            self.push_item(CoverageItem {
432                                kind: CoverageItemKind::Branch { branch_id, path_id: 1 },
433                                loc: self.source_location_for(&node.src),
434                                hits: 0,
435                            });
436                        }
437                    }
438                }
439
440                Ok(())
441            }
442            NodeType::BinaryOperation => {
443                self.push_item(CoverageItem {
444                    kind: CoverageItemKind::Statement,
445                    loc: self.source_location_for(&node.src),
446                    hits: 0,
447                });
448
449                // visit left and right expressions
450                // There could possibly a function call in the left or right expression
451                // e.g: callFunc(a) + callFunc(b)
452                if let Some(expr) = node.attribute("leftExpression") {
453                    self.visit_expression(&expr)?;
454                }
455
456                if let Some(expr) = node.attribute("rightExpression") {
457                    self.visit_expression(&expr)?;
458                }
459
460                Ok(())
461            }
462            // Does not count towards coverage
463            NodeType::FunctionCallOptions |
464            NodeType::Identifier |
465            NodeType::IndexAccess |
466            NodeType::IndexRangeAccess |
467            NodeType::Literal |
468            NodeType::YulLiteralValue |
469            NodeType::YulIdentifier => Ok(()),
470            _ => {
471                warn!("unexpected node type, expected an expression: {:?}", node.node_type);
472                Ok(())
473            }
474        }
475    }
476
477    fn visit_block_or_statement(&mut self, node: &Node) -> eyre::Result<()> {
478        match node.node_type {
479            NodeType::Block => self.visit_block(node),
480            NodeType::Break |
481            NodeType::Continue |
482            NodeType::DoWhileStatement |
483            NodeType::EmitStatement |
484            NodeType::ExpressionStatement |
485            NodeType::ForStatement |
486            NodeType::IfStatement |
487            NodeType::InlineAssembly |
488            NodeType::Return |
489            NodeType::RevertStatement |
490            NodeType::TryStatement |
491            NodeType::VariableDeclarationStatement |
492            NodeType::YulVariableDeclaration |
493            NodeType::WhileStatement => self.visit_statement(node),
494            // Skip placeholder statements as they are never referenced in source maps.
495            NodeType::PlaceholderStatement => Ok(()),
496            _ => {
497                warn!("unexpected node type, expected block or statement: {:?}", node.node_type);
498                Ok(())
499            }
500        }
501    }

503    /// Pushes a coverage item to the internal collection, and might push a line item as well.
504    fn push_item(&mut self, item: CoverageItem) {
505        let source_location = &item.loc;
506
507        // Push a line item if we haven't already
508        if matches!(
509            item.kind,
510            CoverageItemKind::Statement |
511                CoverageItemKind::Branch { .. } |
512                CoverageItemKind::SinglePathBranch { .. }
513        ) && self.last_line < source_location.line
514        {
515            self.items.push(CoverageItem {
516                kind: CoverageItemKind::Line,
517                loc: source_location.clone(),
518                hits: 0,
519            });
520            self.last_line = source_location.line;
521        }
522
523        self.items.push(item);
524    }

526    fn source_location_for(&self, loc: &ast::LowFidelitySourceLocation) -> SourceLocation {
527        SourceLocation {
528            source_id: self.source_id,
529            contract_name: self.contract_name.clone(),
530            start: loc.start as u32,
531            length: loc.length.map(|x| x as u32),
532            line: self.source[..loc.start].lines().count(),
533        }
534    }
535}

537/// Helper function to check if a given node contains any statement.
538fn has_statements(node: &Node) -> bool {
539    let statements: Vec<Node> = node.attribute("statements").unwrap_or_default();
540    !statements.is_empty()

543/// [`SourceAnalyzer`] result type.
544#[derive(Debug)]
545pub struct SourceAnalysis {
546    /// A collection of coverage items.
547    pub items: Vec<CoverageItem>,
548}
549
550/// Analyzes a set of sources to find coverage items.
551#[derive(Debug)]
552pub struct SourceAnalyzer<'a> {
553    sources: &'a SourceFiles<'a>,
554}
555
556impl<'a> SourceAnalyzer<'a> {
557    /// Creates a new source analyzer.
558    pub fn new(data: &'a SourceFiles<'a>) -> Self {
559        Self { sources: data }
560    }
561
562    /// Analyzes contracts in the sources held by the source analyzer.
563    ///
564    /// Coverage items are found by:
565    /// - Walking the AST of each contract (except interfaces)
566    /// - Recording the items of each contract
567    ///
568    /// Each coverage item contains relevant information to find opcodes corresponding to them: the
569    /// source ID the item is in, the source code range of the item, and the contract name the item
570    /// is in.
571    ///
572    /// Note: Source IDs are only unique per compilation job; that is, a code base compiled with
573    /// two different solc versions will produce overlapping source IDs if the compiler version is
574    /// not taken into account.
575    pub fn analyze(&self) -> eyre::Result<SourceAnalysis> {
576        let items = self
577            .sources
578            .sources
579            .par_iter()
580            .flat_map_iter(|(&source_id, SourceFile { source, ast })| {
581                ast.nodes.iter().map(move |node| {
582                    if !matches!(node.node_type, NodeType::ContractDefinition) {
583                        return Ok(vec![]);
584                    }
585
586                    // Skip interfaces which have no function implementations.
587                    let contract_kind: String = node
588                        .attribute("contractKind")
589                        .ok_or_else(|| eyre::eyre!("Contract has no kind"))?;
590                    if contract_kind == "interface" {
591                        return Ok(vec![]);
592                    }
593
594                    let name = node
595                        .attribute("name")
596                        .ok_or_else(|| eyre::eyre!("Contract has no name"))?;
597
598                    let mut visitor = ContractVisitor::new(source_id, source, &name);
599                    visitor.visit_contract(node)?;
600                    let mut items = visitor.items;
601
602                    let is_test = items.iter().any(|item| {
603                        if let CoverageItemKind::Function { name } = &item.kind {
604                            name.is_any_test()
605                        } else {
606                            false
607                        }
608                    });
609                    if is_test {
610                        items.clear();
611                    }
612
613                    Ok(items)
614                })
615            })
616            .collect::<eyre::Result<Vec<Vec<_>>>>()?;
617        Ok(SourceAnalysis { items: items.concat() })
618    }
619}
