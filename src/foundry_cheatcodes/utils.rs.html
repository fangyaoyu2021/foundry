168impl Cheatcode for randomUint_1Call {
169    fn apply(&self, _state: &mut Cheatcodes) -> Result {
170        let Self { min, max } = *self;
171        ensure!(min <= max, "min must be less than or equal to max");
172        // Generate random between range min..=max
173        let mut rng = rand::thread_rng();
174        let exclusive_modulo = max - min;
175        let mut random_number = rng.gen::<U256>();
176        if exclusive_modulo != U256::MAX {
177            let inclusive_modulo = exclusive_modulo + U256::from(1);
178            random_number %= inclusive_modulo;
179        }
180        random_number += min;
181        Ok(random_number.abi_encode())
182    }
183}

193/// Using a given private key, return its public ETH address, its public key affine x and y
194/// coordinates, and its private key (see the 'Wallet' struct)
195///
196/// If 'label' is set to 'Some()', assign that label to the associated ETH address in state
197fn create_wallet(private_key: &U256, label: Option<&str>, state: &mut Cheatcodes) -> Result {
198    let key = parse_private_key(private_key)?;
199    let addr = alloy_signer::utils::secret_key_to_address(&key);
200
201    let pub_key = key.verifying_key().as_affine().to_encoded_point(false);
202    let pub_key_x = U256::from_be_bytes((*pub_key.x().unwrap()).into());
203    let pub_key_y = U256::from_be_bytes((*pub_key.y().unwrap()).into());
204
205    if let Some(label) = label {
206        state.labels.insert(addr, label.into());
207    }
208
209    Ok(Wallet { addr, publicKeyX: pub_key_x, publicKeyY: pub_key_y, privateKey: *private_key }
210        .abi_encode())
211}

213pub(super) fn encode_full_sig(sig: alloy_primitives::Signature) -> Vec<u8> {
214    // Retrieve v, r and s from signature.
215    let v = U256::from(sig.v().y_parity_byte_non_eip155().unwrap_or(sig.v().y_parity_byte()));
216    let r = B256::from(sig.r());
217    let s = B256::from(sig.s());
218    (v, r, s).abi_encode()
219}
220
221pub(super) fn encode_compact_sig(sig: alloy_primitives::Signature) -> Vec<u8> {
222    // Implement EIP-2098 compact signature.
223    let r = B256::from(sig.r());
224    let mut vs = sig.s();
225    vs.set_bit(255, sig.v().y_parity());
226    (r, vs).abi_encode()
227}

229pub(super) fn sign(private_key: &U256, digest: &B256) -> Result<alloy_primitives::Signature> {
230    // The `ecrecover` precompile does not use EIP-155. No chain ID is needed.
231    let wallet = parse_wallet(private_key)?;
232    let sig = wallet.sign_hash_sync(digest)?;
233    debug_assert_eq!(sig.recover_address_from_prehash(digest)?, wallet.address());
234    Ok(sig)
235}

237pub(super) fn sign_with_wallet<DB: DatabaseExt>(
238    ccx: &mut CheatsCtxt<DB>,
239    signer: Option<Address>,
240    digest: &B256,
241) -> Result<alloy_primitives::Signature> {
242    let Some(script_wallets) = ccx.state.script_wallets() else {
243        bail!("no wallets are available");
244    };
245
246    let mut script_wallets = script_wallets.inner.lock();
247    let maybe_provided_sender = script_wallets.provided_sender;
248    let signers = script_wallets.multi_wallet.signers()?;
249
250    let signer = if let Some(signer) = signer {
251        signer
252    } else if let Some(provided_sender) = maybe_provided_sender {
253        provided_sender
254    } else if signers.len() == 1 {
255        *signers.keys().next().unwrap()
256    } else {
257        bail!("could not determine signer");
258    };
259
260    let wallet = signers
261        .get(&signer)
262        .ok_or_else(|| fmt_err!("signer with address {signer} is not available"))?;
263
264    let sig = foundry_common::block_on(wallet.sign_hash(digest))?;
265    debug_assert_eq!(sig.recover_address_from_prehash(digest)?, signer);
266    Ok(sig)
267}

269pub(super) fn sign_p256(private_key: &U256, digest: &B256, _state: &mut Cheatcodes) -> Result {
270    ensure!(*private_key != U256::ZERO, "private key cannot be 0");
271    let n = U256::from_limbs(*p256::NistP256::ORDER.as_words());
272    ensure!(
273        *private_key < n,
274        format!("private key must be less than the secp256r1 curve order ({})", n),
275    );
276    let bytes = private_key.to_be_bytes();
277    let signing_key = P256SigningKey::from_bytes((&bytes).into())?;
278    let signature: Signature = signing_key.sign_prehash(digest.as_slice())?;
279    let r_bytes: [u8; 32] = signature.r().to_bytes().into();
280    let s_bytes: [u8; 32] = signature.s().to_bytes().into();
281
282    Ok((r_bytes, s_bytes).abi_encode())
283}

285pub(super) fn parse_private_key(private_key: &U256) -> Result<SigningKey> {
286    ensure!(*private_key != U256::ZERO, "private key cannot be 0");
287    ensure!(
288        *private_key < U256::from_limbs(*Secp256k1::ORDER.as_words()),
289        "private key must be less than the secp256k1 curve order \
290         (115792089237316195423570985008687907852837564279074904382605163141518161494337)",
291    );
292    let bytes = private_key.to_be_bytes();
293    SigningKey::from_bytes((&bytes).into()).map_err(Into::into)
294}

296pub(super) fn parse_wallet(private_key: &U256) -> Result<PrivateKeySigner> {

300fn derive_key_str(mnemonic: &str, path: &str, index: u32, language: &str) -> Result {
301    match language {
302        "chinese_simplified" => derive_key::<ChineseSimplified>(mnemonic, path, index),
303        "chinese_traditional" => derive_key::<ChineseTraditional>(mnemonic, path, index),
304        "czech" => derive_key::<Czech>(mnemonic, path, index),
305        "english" => derive_key::<English>(mnemonic, path, index),
306        "french" => derive_key::<French>(mnemonic, path, index),
307        "italian" => derive_key::<Italian>(mnemonic, path, index),
308        "japanese" => derive_key::<Japanese>(mnemonic, path, index),
309        "korean" => derive_key::<Korean>(mnemonic, path, index),
310        "portuguese" => derive_key::<Portuguese>(mnemonic, path, index),
311        "spanish" => derive_key::<Spanish>(mnemonic, path, index),
312        _ => Err(fmt_err!("unsupported mnemonic language: {language:?}")),
313    }
314}

316fn derive_key<W: Wordlist>(mnemonic: &str, path: &str, index: u32) -> Result {
317    fn derive_key_path(path: &str, index: u32) -> String {
318        let mut out = path.to_string();
319        if !out.ends_with('/') {
320            out.push('/');
321        }
322        out.push_str(&index.to_string());
323        out
324    }
325
326    let wallet = MnemonicBuilder::<W>::default()
327        .phrase(mnemonic)
328        .derivation_path(derive_key_path(path, index))?
329        .build()?;
330    let private_key = U256::from_be_bytes(wallet.credential().to_bytes().into());
331    Ok(private_key.abi_encode())
332}

361        let mut cheats = cheats();

366        let verifying_key = VerifyingKey::from(&signing_key);
