6/// Mocked call data.
7#[derive(Clone, Debug, Default, PartialEq, Eq, Hash)]
8pub struct MockCallDataContext {
9    /// The partial calldata to match for mock
10    pub calldata: Bytes,
11    /// The value to match for mock
12    pub value: Option<U256>,
13}

15/// Mocked return data.
16#[derive(Clone, Debug)]
17pub struct MockCallReturnData {
18    /// The return type for the mocked call
19    pub ret_type: InstructionResult,
20    /// Return data or error
21    pub data: Bytes,
22}

24impl PartialOrd for MockCallDataContext {
25    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
26        Some(self.cmp(other))
27    }
28}
29
30impl Ord for MockCallDataContext {
31    fn cmp(&self, other: &Self) -> Ordering {
32        // Calldata matching is reversed to ensure that a tighter match is
33        // returned if an exact match is not found. In case, there is
34        // a partial match to calldata that is more specific than
35        // a match to a msg.value, then the more specific calldata takes
36        // precedence.
37        self.calldata.cmp(&other.calldata).reverse().then(self.value.cmp(&other.value).reverse())
38    }
39}

49impl Cheatcode for mockCall_0Call {
50    fn apply_stateful<DB: DatabaseExt>(&self, ccx: &mut CheatsCtxt<DB>) -> Result {
51        let Self { callee, data, returnData } = self;
52        let (acc, _) = ccx.ecx.load_account(*callee)?;
53
54        // Etches a single byte onto the account if it is empty to circumvent the `extcodesize`
55        // check Solidity might perform.
56        let empty_bytecode = acc.info.code.as_ref().map_or(true, Bytecode::is_empty);
57        if empty_bytecode {
58            let code = Bytecode::new_raw(Bytes::from_static(&[0u8]));
59            ccx.ecx.journaled_state.set_code(*callee, code);
60        }
61
62        mock_call(ccx.state, callee, data, None, returnData, InstructionResult::Return);
63        Ok(Default::default())
64    }
65}

92#[allow(clippy::ptr_arg)] // Not public API, doesn't matter
93fn mock_call(
94    state: &mut Cheatcodes,
95    callee: &Address,
96    cdata: &Bytes,
97    value: Option<&U256>,
98    rdata: &Bytes,
99    ret_type: InstructionResult,
100) {
101    state.mocked_calls.entry(*callee).or_default().insert(
102        MockCallDataContext { calldata: Bytes::copy_from_slice(cdata), value: value.copied() },
103        MockCallReturnData { ret_type, data: Bytes::copy_from_slice(rdata) },
104    );
